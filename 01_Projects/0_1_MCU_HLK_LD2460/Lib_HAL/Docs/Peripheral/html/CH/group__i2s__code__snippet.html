Initialize

I2S_Handle_T I2S_Handle = {
/* link to SPI0 IP */
.instance = SN_SPI0,
.init = {
.mode = I2S_MODE_MASTER,
.standard = I2S_STANDARD_MSB,
.channel = I2S_CHANNEL_MONO,
.data_format = I2S_DATA_LEN_8B,
.channel_len = 16,
.audio_freq = I2S_AUDIO_FREQ_48K,
.cpol = I2S_CPOL_LOW,
},
};
/*
* 1 initialize the I2S_Handle with I2S0
* 2 call I2S0 NVIC_EnableIRQ in file msp_i2s.c
* 3 implement I2S0_IRQHandler in file it_i2s.c
*/
if (HAL_I2S_Init(&I2S_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_i2s.c////////////////////////
/**
* @brief I2S MSP Initialization
* @param hi2s: I2S handle pointer
* @retval None
*/
void HAL_I2S_MspInit(I2S_Handle_T *hi2s)
{
NVIC_ClearPendingIRQ(I2S0_IRQn);
NVIC_EnableIRQ(I2S0_IRQn);
}
////////////////////////it_i2s.c////////////////////////
/**
* @brief user should implement the SPI0_IRQHandler for handling I2S0 IRQ.
* By invoking HAL_I2S_IRQHandler with I2S_Handle, the I2S HAL driver will dispatch the IRQ result
* to weak functions(public in I2S HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void SPI0_IRQHandler(void)
{
/* Call I2S IRQ handle */
HAL_I2S_IRQHandler(&I2S_Handle);
}

Deinitialize

/*
* 1 deinitialize the I2S_Handle
* 2 call I2S0 NVIC_DisableIRQ in file msp_i2s.c
*/
if (HAL_I2S_DeInit(&I2S_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_i2s.c////////////////////////
/**
* @brief I2S MSP De-Initialization
* @param hi2s: I2S handle pointer
* @retval None
*/
void HAL_I2S_MspDeInit(I2S_Handle_T *hi2s)
{
NVIC_ClearPendingIRQ(I2S0_IRQn);
NVIC_DisableIRQ(I2S0_IRQn);
}

* @retval None

}

* @brief call HAL_I2S_RegisterCallback to register _tx_done as HAL_I2S_TX_COMPLETE_CB_ID callback function

*/
