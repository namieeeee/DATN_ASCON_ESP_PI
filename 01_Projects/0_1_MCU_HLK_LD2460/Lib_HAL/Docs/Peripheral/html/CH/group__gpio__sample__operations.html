该章节介绍如何使用 GPIO 接口操作引脚。

范例：

#include "sn34f78x_hal.h"
int32_t main(void)
{
GPIO_Init_T GPIO_Init = {0};
/**
* define the GPIO init struct to process all the work flow.
*
*/
/* Set P1.10 as output pin */
GPIO_Init.pin = GPIO_PIN_10;
GPIO_Init.mode = GPIO_MODE_OUTPUT;
HAL_GPIO_Init(SN_GPIO1, &GPIO_Init);
/* Set P1.5 as input pin and IRQ condition is rising edge */
GPIO_Init.pin = GPIO_PIN_5;
GPIO_Init.mode = GPIO_MODE_IT_RISING;
GPIO_Init.pull = GPIO_PULL_DOWN;
HAL_GPIO_Init(SN_GPIO1, &GPIO_Init);
uint32_t pin_value = 0;
/**
* read & write pin data
* 1. set output to high
* 2. read data from input
* 3. check data
*/
HAL_GPIO_WritePin(SN_GPIO1, GPIO_PIN_10, GPIO_PIN_HIGH);
pin_value = HAL_GPIO_ReadPin(SN_GPIO1, GPIO_PIN_5);
if (pin_value != GPIO_PIN_HIGH)
{
return HAL_ERROR;
}
/**
* toggle pin data
* 1. set output to low
* 2. toggle output data
* 3. read data from input
* 4. check whether the value is high
*/
HAL_GPIO_WritePin(SN_GPIO1, GPIO_PIN_10, GPIO_PIN_LOW);
HAL_GPIO_TogglePin(SN_GPIO1, GPIO_PIN_10);
pin_value = HAL_GPIO_ReadPin(SN_GPIO1, GPIO_PIN_5);
if (pin_value != GPIO_PIN_HIGH)
{
return HAL_ERROR;
}
/* deinit gpio */
HAL_GPIO_DeInit(SN_GPIO1, GPIO_PIN_5 | GPIO_PIN_10);
while (true)
{
__NOP();
}
}
