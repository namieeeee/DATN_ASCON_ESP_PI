OC Channel Mode
TPM Interface » Test Sample

该章节介绍了如何使用TPM OC channel 模式

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
#include "main.h"
static uint32_t ch0_flag = 0;
TPM_Handle_T TPM0_Handle = {
.instance = (void *)SN_CT16B0,
.init = {
.pre_scaler = 47,
.mode = TPM_CTM_TIMER,
.cm = TPM_COUNTING_UP,
},
};
TPM_OC_T CH0 = {
.mode = TPM_OCMODE_TIMER,
.period = 3000,
.stop = HAL_DISABLE,
.reset = HAL_DISABLE,
};
void HAL_TPM_OC_DelayElapsedCallback(TPM_Handle_T *htpm)
{
ch0_flag = 1;
}
int32_t tpm_oc(void)
{
if (HAL_TPM_Init(&TPM0_Handle) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_OC_ConfigChannel(&TPM0_Handle, &CH0, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_OC_Start_IT(&TPM0_Handle, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
while (ch0_flag == 0)
;
if (HAL_TPM_OC_Stop_IT(&TPM0_Handle, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_DeInit(&TPM0_Handle) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
tpm_oc();
while (true)
{
__NOP();
}
}
