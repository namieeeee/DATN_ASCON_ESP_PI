该章节介绍如何使用 TPM PWM 和 PWM 输出

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
static uint32_t ch0_flag = 0;
TPM_Handle_T TPM0_Handle = {
.instance = (void *)SN_CT16B0,
.init = {
.pre_scaler = 47,
.mode = TPM_CTM_TIMER,
.cm = TPM_COUNTING_UP,
},
};
TPM_OC_T CH0 = {
.mode = TPM_OCMODE_TIMER,
.period = 500,
.stop = HAL_DISABLE,
.reset = HAL_DISABLE,
};
TPM_OC_T CH9 = {
.mode = TPM_OCMODE_TIMER,
.period = 1000,
.stop = HAL_DISABLE,
.reset = HAL_ENABLE,
};
void HAL_TPM_PWM_PulseFinishedCallback(TPM_Handle_T *htpm)
{
ch0_flag = 1;
}
int32_t tpm_pwm(void)
{
TPM_PWMN_T DB0 = {
.mode = TPM_PWMN_MODE_0,
.db = 10,
};
if (HAL_TPM_Init(&TPM0_Handle) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_PWM_ConfigChannel(&TPM0_Handle, &CH0, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_PWM_ConfigPWMN(&TPM0_Handle, &DB0, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
// CH9 for reset TC
if (HAL_TPM_OC_ConfigChannel(&TPM0_Handle, &CH9, TPM_CHANNEL_9) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_PWM_Start_IT(&TPM0_Handle, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
while (ch0_flag == 0)
;
if (HAL_TPM_PWM_Stop_IT(&TPM0_Handle, TPM_CHANNEL_0) != HAL_OK)
return HAL_ERROR;
if (HAL_TPM_DeInit(&TPM0_Handle) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
tpm_pwm();
while (true)
{
__NOP();
}
}
