Initialize

UART_Handle_T UART_Handle = {
/* link to UART0 IP */
.instance = SN_UART0,
.init = {
.baud_rate = 115200,
.word_length = UART_WORD_LENGTH_8,
.stop_bits = UART_STOP_BITS_1,
.parity = UART_PARITY_NONE,
},
};
/*
* 1 initialize the UART_Handle with UART0
* 2 call UART0 NVIC_EnableIRQ in file msp_uart.c
* 3 implement UART0_IRQHandler in file it_uart.c
*/
if (HAL_UART_Init(&UART_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_uart.c////////////////////////
/**
* @brief UART MSP Initialization
* @param huart: UART handle pointer
* @retval None
*/
void HAL_UART_MspInit(UART_Handle_T *huart)
{
NVIC_ClearPendingIRQ(UART0_IRQn);
NVIC_EnableIRQ(UART0_IRQn);
}
////////////////////////it_uart.c////////////////////////
/**
* @brief user should implement the UART0_IRQHandler for handling UART0 IRQ.
* By invoking HAL_UART_IRQHandler with UART_Handle, the UART HAL driver will dispatch the IRQ result
* to weak functions(public in UART HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void UART0_IRQHandler(void)
{
/* Call UART IRQ handle */
HAL_UART_IRQHandler(&UART_Handle);
}

Deinitialize

/*
* 1 deinitialize the UART_Handle
* 2 call UART0 NVIC_DisableIRQ in file msp_uart.c
*/
if (HAL_UART_DeInit(&UART_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_uart.c////////////////////////
/**
* @brief UART MSP De-Initialization
* @param huart: UART handle pointer
* @retval None
*/
void HAL_UART_MspDeInit(UART_Handle_T *huart)
{
NVIC_ClearPendingIRQ(UART0_IRQn);
NVIC_DisableIRQ(UART0_IRQn);
}
