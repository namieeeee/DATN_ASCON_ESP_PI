Blocking Mode
I2S Interface » Test Sample

该章节介绍了如何在阻塞模式下使用I2S接口传输数据。

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 32
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/* timeout ticks */
static uint32_t timeout = HAL_MAX_DELAY;
/*****************************************************************************/
static int32_t _i2s_blocking_transmit(void)
{
I2S_Handle_T i2s_handle = {
.instance = SN_SPI0,
.init = {
.mode = I2S_MODE_MASTER,
.standard = I2S_STANDARD_MSB,
.channel = I2S_CHANNEL_MONO,
.data_format = I2S_DATA_LEN_8B,
.channel_len = 16,
.audio_freq = I2S_AUDIO_FREQ_48K,
.cpol = I2S_CPOL_LOW,
},
};
if (HAL_I2S_Init(&i2s_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2S_Transmit(&i2s_handle, data_tx, DATA_LEN, timeout) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2S_DeInit(&i2s_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
static int32_t _i2s_blocking_receive(void)
{
I2S_Handle_T i2s_handle = {
.instance = SN_SPI1,
.init = {
.mode = I2S_MODE_MASTER,
.standard = I2S_STANDARD_MSB,
.channel = I2S_CHANNEL_MONO,
.data_format = I2S_DATA_LEN_8B,
.channel_len = 16,
.audio_freq = I2S_AUDIO_FREQ_48K,
.cpol = I2S_CPOL_LOW,
},
};
if (HAL_I2S_Init(&i2s_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2S_Receive(&i2s_handle, data_rx, DATA_LEN, timeout) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2S_DeInit(&i2s_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_i2s_blocking_transmit();
_i2s_blocking_receive();
while (true)
{
__NOP();
}
}
