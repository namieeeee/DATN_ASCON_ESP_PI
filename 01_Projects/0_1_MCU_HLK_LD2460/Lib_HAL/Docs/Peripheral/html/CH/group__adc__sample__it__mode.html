Interrupt Mode
ADC Interface » Test Sample

该章节介绍了如何使用 ADC 接口在 Interrupt 模式转换电压

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
static uint32_t data = 0;
static uint16_t conv_flag = 0;
ADC_Handle_T ADC_Handle = {
ADC_Handle.instance = SN_ADC0,
.init = {
.conv_mode = ADC_CONVMODE_SINGLE_STEP,
.pre_clock = 5,
},
};
void HAL_ADC_ConvCpltCallback(ADC_Handle_T *hadc)
{
data = HAL_ADC_GetValue(hadc, ADC_CHANNEL_1);
conv_flag = 1;
}
uint32_t adc_it_start(void)
{
if (HAL_ADC_Init(&ADC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
// Input 3.3V to P2.1
ADC_Channel_T ADC_CH1 = {
.channel = ADC_CHANNEL_1,
.rank = 1,
};
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH1) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_SelectRefOption(&ADC_Handle, ADC_REF_VDD_GND) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_Start_IT(&ADC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
while (!conv_flag)
;
if (data != 0xFFF)
{
return HAL_ERROR;
}
if (HAL_ADC_DeInit(&ADC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
adc_it_start();
while (true)
{
__NOP();
}
}
