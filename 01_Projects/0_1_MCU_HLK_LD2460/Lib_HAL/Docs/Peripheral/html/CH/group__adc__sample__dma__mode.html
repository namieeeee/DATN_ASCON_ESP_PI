DMA Mode
ADC Interface » Test Sample

该章节介绍了如何使用 ADC 接口在 DMA 模式转换电压，传输数据

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
#include "main.h"
static uint32_t data[20] = {0};
static uint16_t conv_flag = 0;
DMA_Handle_T DMA_ADC = {
.instance = SN_DMA1,
.init = {
.channel = DMA_CHANNEL_0,
.ch_xfer_mode = DMA_PERIPH_TO_MEMORY,
.ch_priority = DMA_PRIORITY_VERY_HIGH,
.ch_fifo_th = DMA_FIFO_TH_1,
.src_rs = DMA_RS1_ADC0_0,
.src_burst = DMA_SRC_BURST_1,
.src_width = DMA_XFR_WIDTH_32BITS, // do 32bits data transfer
.src_addr_mode = DMA_ADDR_MODE_FIXED, // current use ADC data register, so the address mode need set to FIXED
.src_ahb_master = DMA_AHB_MASTER_PERIPH,
.dst_width = DMA_XFR_WIDTH_32BITS, // do 32bits data transfer
.dst_addr_mode = DMA_ADDR_MODE_CYCLIC, // the destination address should increase
.dst_ahb_master = DMA_AHB_MASTER_PERIPH,
}};
ADC_Handle_T ADC_Handle = {
ADC_Handle.instance = SN_ADC0,
.init = {
.conv_mode = ADC_CONVMODE_LOOP_SCAN,
.pre_clock = 5,
.scan_ch_num = 4,
},
};
void ADC_DMA_Loop_ConvCpltCallback(ADC_Handle_T *hadc)
{
conv_flag = 1;
}
uint32_t adc_dma_start(void)
{
if (HAL_DMA_Init(&DMA_ADC) != HAL_OK)
{
return HAL_ERROR;
}
HAL_LINKDMA(&ADC_Handle, hdma[ADC_DMA_CHANNEL_0], DMA_ADC);
if (HAL_ADC_Init(&ADC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_RegisterCallback(&ADC_Handle, HAL_ADC_CONV_CPLT_CB_ID, ADC_DMA_Loop_ConvCpltCallback) != HAL_OK)
{
return HAL_ERROR;
}
ADC_Channel_T ADC_CH0 = {
.channel = ADC_CHANNEL_0,
.rank = 1,
};
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH0) != HAL_OK)
{
return HAL_ERROR;
}
ADC_Channel_T ADC_CH1 = {
.channel = ADC_CHANNEL_1,
.rank = 2,
};
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH1) != HAL_OK)
{
return HAL_ERROR;
}
ADC_Channel_T ADC_CH2 = {
.channel = ADC_CHANNEL_2,
.rank = 3,
};
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH2) != HAL_OK)
{
return HAL_ERROR;
}
ADC_Channel_T ADC_CH3 = {
.channel = ADC_CHANNEL_3,
.rank = 4,
};
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH3) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_ConfigChannel(&ADC_Handle, &ADC_CH_VSS) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_SelectRefOption(&ADC_Handle, ADC_REF_VDD_GND) != HAL_OK)
{
return HAL_ERROR;
}
ADC_DMA_Channel_T ADC_DMA_CH0 = {
.dma_channel = ADC_DMA_CHANNEL_0,
.injected_mode = ADC_DMA_INJECT_ALL_CONV,
.fifo_threshold = ADC_DMA_FIFO_4,
};
if (HAL_ADC_ConfigDMA_Channel(&ADC_Handle, &ADC_DMA_CH0) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_ADC_Start_DMA(&ADC_Handle, ADC_DMA_CHANNEL_0, data, 18 * 5) != HAL_OK)
{
return HAL_ERROR;
}
while (!conv_flag)
;
if (HAL_ADC_DeInit(&ADC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
adc_dma_start();
while (true)
{
__NOP();
}
}
