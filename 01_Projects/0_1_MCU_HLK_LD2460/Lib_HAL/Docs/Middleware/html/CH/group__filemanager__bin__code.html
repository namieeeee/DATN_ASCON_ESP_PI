#include "sn34f78x_hal.h"
#define BIN_SIZE 1024
// The Sample function is to read a bin file(test_bin_r.bin) and then write the read content to another bin file(test_bin_w.bin)
HAL_Status_T bin_rw_block(void)
{
// The file being read
M_FILE_HANDLE_T fh_R = {0};
M_FILE_INIT_T initR = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is bin file, it is FILE_TYPE_BIN
.format = FILE_TYPE_BIN,
// Open File Method
.mode = FILE_OPEN_R,
// The file path, excluding drive letters
.full_path = "test_bin_r.bin",
};
// The file being write
M_FILE_HANDLE_T fh_W = {0};
M_FILE_INIT_T initW = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is bin file, it is FILE_TYPE_BIN
.format = FILE_TYPE_BIN,
// Open File Method
.mode = FILE_OPEN_WX,
// The file path, excluding drive letters
.full_path = "test_bin_w.bin",
};
// The file content cache area
M_FILE_INFO_T r_data = {0};
uint8_t *buffer = pvPortMalloc(BIN_SIZE);
if (buffer == NULL)
{
goto __test_err__;
}
r_data.cache = buffer;
r_data.cache_len = BIN_SIZE;
// Open file
if (HAL_MFile_OpenFile(&fh_R, &initR) != HAL_OK)
goto __test_err__;
if (HAL_MFile_OpenFile(&fh_W, &initW) != HAL_OK)
goto __test_err__;
while (fh_R.bin_eof == M_FILE_IS_NOT_EOF)
{
// Read data from file and save content to r_data
if (HAL_MFile_ReadFile(&fh_R, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
// Write data to file
if (HAL_MFile_WriteFile(&fh_W, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
}
if (buffer)
{
vPortFree(buffer);
}
//Close file
if (HAL_MFile_CloseFile(fh_W) != HAL_OK)
goto __test_err__;
//Close file
if (HAL_MFile_CloseFile(fh_R) != HAL_OK)
goto __test_err__;
return HAL_OK;
__test_err__:
return HAL_ERROR;
}
