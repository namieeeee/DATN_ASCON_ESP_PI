Initialize

IWDG_Handle_T IWDG_handle = {
.instance = SN_WDT,
.init.clk_source = IWDG_CLK_SOURCE_ILRC,
.init.mode = IWDG_MODE_IRQ,
.init.timer_constant = 0x000FFFFF,
};
/*
* 1 initialize the IWDG_handle
* 2 call WDT NVIC_EnableIRQ in file msp_iwdg.c
* 3 implement WDT_IRQHandler in file it_iwdg.c
*/
if (HAL_IWDG_Init(&IWDG_handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_iwdg.c////////////////////////
/**
* @brief IWDG MSP Initialization
* @param hiwdg: IWDG handle pointer
* @retval None
*/
void HAL_IWDG_MspInit(IWDG_Handle_T *hiwdg)
{
NVIC_ClearPendingIRQ(WDT_IRQn);
NVIC_EnableIRQ(WDT_IRQn);
}
////////////////////////it_iwdg.c////////////////////////
/**
* @brief user should implement the WDT_IRQHandler for handling IWDG IRQ.
* By invoking HAL_IWDG_IRQHandle with IWDG_Handle, the IWDG HAL driver will dispatch the IRQ result
* to weak functions(public in IWDG HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void WDT_IRQHandler(void)
{
/* Call IWDG IRQ handle */
HAL_IWDG_IRQHandle(&IWDG_handle);
}

Deinitialize

/*
* 1 deinitialize the IWDG_Handle
* 2 call WDT NVIC_DisableIRQ in file msp_iwdg.c
*/
if (HAL_IWDG_DeInit(&IWDG_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_iwdg.c////////////////////////
/**
* @brief IWDG MSP De-Initialization
* @param hiwdg: IWDG handle pointer
* @retval None
*/
void HAL_IWDG_MspDeInit(IWDG_Handle_T *hiwdg)
{
NVIC_ClearPendingIRQ(WDT_IRQn);
NVIC_DisableIRQ(WDT_IRQns);
}
