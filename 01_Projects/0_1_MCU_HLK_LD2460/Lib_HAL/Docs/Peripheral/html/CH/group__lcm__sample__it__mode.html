该 Sample 介绍了如何使用LCM接口清除背景

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
#define RED 0xF800
#define GREEN 0x07E0
uint32_t cur_color = 0;
/**
* @brief Panel displays one color under dma mode flow:
* > 1 initialize panel
* > 2 clear the LCD with red color(panel size : 480*800 = 384000)
* > 3 display in red or green for 1 second
* @retval HAL status
*/
HAL_Status_T lcd_clear(void)
{
// 0x3C00 only for NT35510
if (HAL_LCM_WriteCmd(&LCM_Handle, 0x3C00) != HAL_OK)
return HAL_ERROR;
xfer_num = 384000; // LCD resolution is 480 * 800 = 384000
for (uint32_t i = 0; i < xfer_num; i++)
{
HAL_LCM_WriteData(&LCM_Handle, *color);
}
return HAL_OK;
}
HAL_Status_T switch_background_color()
{
cur_color = RED;
// 1 initialize panel
if (panel_init() != HAL_OK)
return HAL_ERROR;
while (1)
{
if (lcd_clear() != HAL_OK)
return HAL_ERROR;
if (cur_color == RED)
cur_color = GREEN;
else
cur_color = RED;
HAL_Delay(1000);
}
}
/*****************************************************************************/
int32_t main(void)
{
switch_background_color();
while (true)
{
__NOP();
}
}
