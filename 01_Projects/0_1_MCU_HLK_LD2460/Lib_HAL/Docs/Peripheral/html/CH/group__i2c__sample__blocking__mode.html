Blocking Mode
I2C Interface » Test Sample

该章节介绍了如何在 Blocking 模式下使用 I2C 接口传输数据

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 32
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/*****************************************************************************/
I2C_Handle_T i2c_tx_handle = {
/* link to I2C1 IP */
.instance = SN_I2C1,
.init = {
.addressing_mode = I2C_ADDRESSINGMODE_7BIT,
.clock_speed = I2C_CLOCKSPEED_FAST_PLUS,
.gc_en = HAL_DISABLE,
.own_address = 0x7f,
},
};
static int32_t _i2c_blocking_transmit(void)
{
if (HAL_I2C_Init(&i2c_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
// timeout = 6000ms
if (HAL_I2C_Master_Transmit(&i2c_tx_handle, i2c_tx_handle.init.own_address, data_tx, DATA_LEN, 6000) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_DeInit(&i2c_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
I2C_Handle_T i2c_rx_handle = {
/* link to I2C2 IP */
.instance = SN_I2C2,
.init = {
.addressing_mode = I2C_ADDRESSINGMODE_7BIT,
.clock_speed = I2C_CLOCKSPEED_FAST_PLUS,
.gc_en = HAL_DISABLE,
.own_address = 0x7f,
}};
static int32_t _i2c_blocking_receive(void)
{
if (HAL_I2C_Init(&i2c_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
// timeout = 6000ms
if (HAL_I2C_Slave_Receive(&i2c_rx_handle, data_rx, DATA_LEN, 6000) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_DeInit(&i2c_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_i2c_blocking_transmit();
_i2c_blocking_receive();
while (true)
{
__NOP();
}
}
