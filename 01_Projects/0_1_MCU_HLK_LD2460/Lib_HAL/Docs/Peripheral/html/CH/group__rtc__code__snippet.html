RTC Interface » Usage Flow

Initialize

RTC_Handle_T RTC_Handle = {
.init = {
.clk_source = RTC_CLK_SOURCE_ILRC,
},
};
/*
* 1 initialize the RTC
* 2 call SCU NVIC_EnableIRQ in file msp_rtc.c
* 3 implement SCU_IRQHandler in file it_rtc.c
*/
if (HAL_RTC_Init(&RTC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_rtc.c////////////////////////
/**
* @brief RTC MSP Initialization
* @param hRTC: RTC handle pointer
* @retval None
*/
void HAL_RTC_MspInit(RTC_Handle_T *hRTC)
{
NVIC_ClearPendingIRQ(SCU_IRQn);
NVIC_EnableIRQ(SCU_IRQn);
}
////////////////////////it_rtc.c////////////////////////
/**
* @brief user should implement the SCU_IRQHandler for handling SCU IRQ.
* @return void
*/
__irq void SCU_IRQHandler(void)
{
HAL_SCU_IRQHandler();
}
void HAL_RTC_Callback(uint32_t ris)
{
HAL_RTC_AlarmIRQHandler(&RTC_Handle, ris);
}

Deinitialize

/*
* 1 deinitialize the I2C_Handle
* 2 call SCU NVIC_DisableIRQ in file msp_rtc.c
*/
if (HAL_RTC_DeInit(&RTC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_rtc.c////////////////////////
/**
* @brief RTC MSP De-Initialization
* @param hrtc: RTC handle pointer
* @retval None
*/
void HAL_RTC_MspDeInit(RTC_Handle_T *hrtc)
{
NVIC_ClearPendingIRQ(SCU_IRQn);
NVIC_DisableIRQ(SCU_IRQn);
}

/**

}

*/
