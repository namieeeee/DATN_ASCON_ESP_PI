Initialize

ADC_Handle_T ADC_handle = {
.instance = SN_ADC0,
.init = {
// Specifies the single scan mode
.conv_mode = ADC_CONVMODE_SINGLE_SCAN,
// Specifies the scan conversion channel numbers, ADC scan conversion channel numbers is scan_ch_num + 1 in each scan.
.scan_ch_num = 1,
// Disable auto Power down
.autopwdn = HAL_DISABLE,
// Specifies the prescaler clock (MCLK)
.pre_clock = 5,
},
};
/*
* 1 initialize the ADC_Handle with ADC0
* 2 call ADC0 NVIC_EnableIRQ in file msp_adc.c
* 3 implement ADC0_IRQHandler in file it_adc.c
*/
if (HAL_ADC_Init(&ADC_handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_adc.c////////////////////////
/**
* @brief ADC MSP Initialization
* @param hadc: ADC handle pointer
* @retval None
*/
void HAL_ADC_MspInit(ADC_Handle_T *hadc)
{
NVIC_ClearPendingIRQ(ADC0_IRQn);
NVIC_EnableIRQ(ADC0_IRQn);
}
////////////////////////it_adc.c////////////////////////
/**
* @brief user should implement the ADC0_IRQHandler for handling ADC0 IRQ.
* By invoking HAL_ADC_IRQHandler with ADC_Handle, the ADC HAL driver will dispatch the IRQ result
* to weak functions(public in ADC HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void ADC0_IRQHandler(void)
{
/* Call HAL ADC IRQ handle */
HAL_ADC_IRQHandler(&ADC_handle);
}

Deinitialize

/*
* 1 deinitialize the ADC_Handle
* 2 call ADC0 NVIC_DisableIRQ in file msp_adc.c
*/
if (HAL_ADC_DeInit(&ADC_handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_adc.c////////////////////////
/**
* @brief ADC MSP De-Initialization
* @param hadc: ADC handle pointer
* @retval None
*/
void HAL_ADC_MspDeInit(ADC_Handle_T *hadc)
{
NVIC_ClearPendingIRQ(ADC0_IRQn);
NVIC_DisableIRQ(ADC0_IRQn);
}
