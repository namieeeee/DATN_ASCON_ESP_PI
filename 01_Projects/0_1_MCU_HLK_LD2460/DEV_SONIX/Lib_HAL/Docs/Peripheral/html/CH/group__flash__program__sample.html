该章节介绍了如何使用Flash接口执行编程操作。

范例：

#include "sn34f78x_hal.h"
#define FLASH_USER_START_ADDR 0x40000
#define FLASH_USER_END_ADDR 0x60000
#define FLASH_USER_DATA_SIZE (FLASH_USER_END_ADDR - FLASH_USER_START_ADDR + 1)
#define PAGE_SIZE 512
#define PAGEOFSECTOR 8
uint8_t Buf[PAGE_SIZE];
int ProgramCompletetion = 0;
void HAL_FLASH_EndOfOperationCallback(uint32_t ReturnValue)
{
/* Program complete */
if (ReturnValue == FLASH_USER_START_ADDR)
{
ProgramCompletetion = 1;
}
}
/* Flash irq handler */
__irq void FLASH_IRQHandler(void)
{
HAL_FLASH_IRQHandler();
}
HAL_Status_T Program_Byte_Blocking()
{
uint64_t Data;
Data = (uint8_t)HAL_GetTick();
if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_BYTE, FLASH_USER_START_ADDR, Data) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}

HAL_Status_T Program_Page_Blocking()
{
uint64_t Data;
int i = 0, j = 0;
/* Prepare data buffer for page program */
memset(Buf, 0, PAGE_SIZE);
j = HAL_GetTick();
for (i = 0; i < PAGE_SIZE; i++)
{
Buf[i] = j++;
}
Data = (uint64_t)Buf;
if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_PAGE, FLASH_USER_START_ADDR, Data) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
HAL_Status_T Program_Byte_NonBlocking()
{
uint64_t Data;
Data = (uint8_t)HAL_GetTick();
ProgramCompletetion = 0;
if (HAL_FLASH_Program_IT(FLASH_TYPEPROGRAM_BYTE, FLASH_USER_START_ADDR, Data) != HAL_OK)
{
return HAL_ERROR;
}
while (!ProgramCompletetion)
;
return HAL_OK;
}

HAL_Status_T FMC_Program_Sample(void)
{
/* Program byte */
if (Program_Byte_Blocking() != HAL_OK)
return HAL_ERROR;
/* Program halfword */
if (Program_HalfWord_Blocking() != HAL_OK)
return HAL_ERROR;
/* Program word */
if (Program_Word_Blocking() != HAL_OK)
return HAL_ERROR;
/* Program doubleword */
if (Program_DoubleWord_Blocking() != HAL_OK)
return HAL_ERROR;
/* Program page */
if (Program_Page_Blocking() != HAL_OK)
return HAL_ERROR;
/* Enable IRQ */
NVIC_EnableIRQ(FLASH_IRQn);
/* Program byte in non-blocking mode*/
if (Program_Byte_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Program halfword in non-blocking mode*/
if (Program_HalfWord_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Program word in non-blocking mode*/
if (Program_Word_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Program doubleword in non-blocking mode*/
if (Program_DoubleWord_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Program page in non-blocking mode */
if (Program_Page_NonBlocking() != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
int main(void)
{
uint32_t ret = HAL_OK;
/* HAL Driver Init */
HAL_Init();
/* Configure the system clock */
SystemClock_Config();
/* Flash program sample */
ret = FMC_Program_Sample();
if (ret != HAL_OK)
Error_Handler();
while (1)
;
}
