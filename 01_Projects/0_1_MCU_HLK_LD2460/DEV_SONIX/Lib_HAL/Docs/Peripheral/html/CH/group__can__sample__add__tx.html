Add message to can
CAN Interface » Test Sample

该章节介绍了如何使用CAN接口发送消息至CAN节点。

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 8
/* data buffer for transmitting data */
static uint8_t msg_send[DATA_LEN];
/* data header for transmitting data */
static CAN_TxHeader_T tx_header = {
.id = 0x1FFFFFFF,
.ide = CAN_ID_EXT,
.rtr = CAN_RTR_DATA,
.dlc = 1,
};
/*****************************************************************************/
CAN_Handle_T can0_handle = {
.instance = SN_CAN0,
.init = {
.mode = CAN_MODE_NORMAL,
.pre_scaler = 0, // pre_scaler time_seg1 and time_seg2 Setting CAN0 baudrate to 1M under 48M system clock
.time_seg1 = 44,
.time_seg2 = 1,
.sync_jump_width = 0,
.transmit_fifo_priority = HAL_DISABLE, // STB mailbox transmission using fifo mode
.auto_retransmission = HAL_ENABLE, // choice the non-automatic retransmission mode
},
};
static int32_t _can_add_tx(void)
{
if (HAL_CAN_Init(&can0_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_CAN_Start(&can0_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_CAN_AddTxMessage(&can0_handle, &tx_header, msg_send, CAN_MAILBOX_PTB) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_CAN_Stop(&can0_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_CAN_DeInit(&can0_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_can_add_tx();
while (true)
{
__NOP();
}
}
