该章节介绍如何在 Interrupt 模式下使用 WWDG

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
#define REFRESH_CNT 5
#define TIMEOUT_CNT 5
static __IO uint8_t refresh_cnt = 0;
static __IO uint8_t timeout_cnt = 0;
/*****************************************************************************/
void HAL_WWDG_TimeoutCallback(WWDG_Handle_T *hwwdg)
{
timeout_cnt++;
}
WWDG_Handle_T wwdg_handle = {
.instance = SN_WWDT,
.init.prescaler = WWDG_PRESCALER_4,
.init.counter = 0x10000,
.init.window = 0x5000,
.init.clk_source = WWDG_CLK_SOURCE_ILRC,
.init.mode = WWDG_MODE_OVERFLOW,
};
static int32_t _wwdg_interrupt(void)
{
if (HAL_WWDG_Init(&wwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
while (refresh_cnt < REFRESH_CNT)
{
if (wwdg_handle.instance->RELOAD < wwdg_handle->init.window)
{
if (HAL_WWDG_Refresh(&wwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
refresh_cnt++;
}
}
while (timeout_cnt < TIMEOUT_CNT)
{
__NOP();
}
if (HAL_WWDG_DeInit(&wwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_wwdg_interrupt();
while (true)
{
__NOP();
}
}
