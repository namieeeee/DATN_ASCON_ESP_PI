该章节介绍了如何使用 Flash 接口执行擦除操作。

范例：

#include "sn34f78x_hal.h"
#define FLASH_USER_START_ADDR 0x40000
#define FLASH_USER_END_ADDR 0x60000
#define FLASH_USER_DATA_SIZE (FLASH_USER_END_ADDR - FLASH_USER_START_ADDR + 1)
#define PAGE_SIZE 512
#define PAGEOFSECTOR 8
void HAL_FLASH_EndOfOperationCallback(uint32_t ReturnValue)
{
/* Erase complete */
if (ReturnValue == 0xFFFFFFFF)
{
EraseCompletetion = 1;
}
}
/* Flash irq handler */
__irq void FLASH_IRQHandler(void)
{
HAL_FLASH_IRQHandler();
}
uint8_t Buf[PAGE_SIZE];
uint32_t Flash_Fill_Up()
{
int i, j;
uint8_t *pBuf = NULL;
uint64_t Data;
int size = 512;
uint32_t StartAddress = 0;
/* Program the user Flash area word by word*/
memset(Buf, 0, PAGE_SIZE);
j = HAL_GetTick();
for (i = 0; i < PAGE_SIZE; i++)
{
Buf[i] = j++;
}
pBuf = Buf;
StartAddress = FLASH_USER_START_ADDR;
Data = (uint64_t)pBuf;
/* Fill up MM */
while (StartAddress < FLASH_USER_END_ADDR)
{
if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_PAGE, StartAddress, Data) == HAL_OK)
{
StartAddress += size;
}
else
{
return HAL_FLASH_GetError();
}
}
return 0;
}
HAL_Status_T Mass_Erase_Blocking()
{
FLASH_EraseInitTypeDef EraseInitStruct;
uint32_t Error = 0;
/* Fill in data into flash */
Flash_Fill_Up();
EraseInitStruct.TypeErase = FLASH_TYPEERASE_MASSERASE;
if (HAL_FLASHEx_Erase(&EraseInitStruct, &Error) != HAL_OK)
{
/*Error occurred while page erase.*/
return HAL_ERROR;
}
return HAL_OK;
}
HAL_Status_T Sector_Erase_Blocking()
{
FLASH_EraseInitTypeDef EraseInitStruct;
uint32_t Error = 0;
/* Fill in data into flash */
Flash_Fill_Up();
/* Each sector has 8 pages */
EraseInitStruct.TypeErase = FLASH_TYPEERASE_SECTORS;
EraseInitStruct.Sector = FLASH_USER_START_ADDR / (PAGEOFSECTOR * PAGE_SIZE);
EraseInitStruct.NbSectors = FLASH_USER_DATA_SIZE / (PAGEOFSECTOR * PAGE_SIZE);
if (HAL_FLASHEx_Erase(&EraseInitStruct, &Error) != HAL_OK)
{
/*Error occurred while page erase.*/
return HAL_ERROR;
}
return HAL_OK;
}
HAL_Status_T Page_Erase_Blocking()
{
FLASH_EraseInitTypeDef EraseInitStruct;
uint32_t Error = 0;
/* Fill in data into flash */
Flash_Fill_Up();
EraseInitStruct.TypeErase = FLASH_TYPEERASE_PAGES;
EraseInitStruct.Page = (FLASH_USER_START_ADDR / PAGE_SIZE);
EraseInitStruct.NbPages = (FLASH_USER_DATA_SIZE / PAGE_SIZE);
if (HAL_FLASHEx_Erase(&EraseInitStruct, &Error) != HAL_OK)
{
/*Error occurred while page erase.*/
return HAL_ERROR;
}
return HAL_OK;
}
HAL_Status_T Mass_Erase_NonBlocking()
{
FLASH_EraseInitTypeDef EraseInitStruct;
/* Fill in data into flash */
Flash_Fill_Up();
EraseInitStruct.TypeErase = FLASH_TYPEERASE_MASSERASE;
EraseCompletetion = 0;
if (HAL_FLASHEx_Erase_IT(&EraseInitStruct) != HAL_OK)
{
/*Error occurred while page erase.*/
return HAL_ERROR;
}
while (!EraseCompletetion)
;
return HAL_OK;
}

HAL_Status_T FMC_Erase_Sample(void)
{
/* Copy NVIC Table
* Because Mass Erase will erase entire flash
*/
NVICTable2SRAM(0x10000000, &__Vectors_End - &__Vectors);
/* Mass Erase */
if (Mass_Erase_Blocking() != HAL_OK)
return HAL_ERROR;
/* Sector Erase */
if (Sector_Erase_Blocking() != HAL_OK)
return HAL_ERROR;
/* Page Erase */
if (Page_Erase_Blocking() != HAL_OK)
return HAL_ERROR;
/* Enable IRQ */
NVIC_EnableIRQ(FLASH_IRQn);
/* Mass Erase in non-blocking mode */
if (Mass_Erase_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Sector Erase in non-blocking mode */
if (Sector_Erase_NonBlocking() != HAL_OK)
return HAL_ERROR;
/* Page Erase in non-blocking mode*/
if (Page_Erase_NonBlocking() != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
int main(void)
{
uint32_t ret = HAL_OK;
/* HAL Driver Init */
HAL_Init();
/* Configure the system clock */
SystemClock_Config();
/* Flash program sample */
ret = FMC_Erase_Sample();
if (ret != HAL_OK)
Error_Handler();
while (1)
;
}
