#include "sn34f78x_hal.h"
NORFlash_Handle_T NORFlash_Handle;
uint32_t sn34f2_NORFlash_Init(void)
{
NORFlash_Handle.init.instance = SN_SPI0;
NORFlash_Handle.init.fs_polarity = NOR_FS_POLARITY_LOW;
NORFlash_Handle.init.clk_polarity = NOR_CLK_POLARITY_LOW;
NORFlash_Handle.init.clk_phase = NOR_CLK_PHASE_1EDGE;
NORFlash_Handle.init.sclk_div = 3;
NORFlash_Handle.init.first_bit = NOR_FIRST_BIT_MSB;
NORFlash_Handle.init.cs_gpio_port = SN_GPIO2;
NORFlash_Handle.init.cs_gpio_pin = GPIO_PIN_0;
if (HAL_OK != HAL_NORFlash_Init(&NORFlash_Handle))
{
return HAL_ERROR;
}
return HAL_OK;
}
void fill_data(uint8_t *data, uint32_t cnt, uint16_t start, uint16_t data_bit)
{
uint32_t bit_mask = 0xffff;
bit_mask >>= (16 - data_bit);
for (uint16_t i = 0; i < program_size; i++)
{
if (i < cnt)
data[i] = (i + start) & bit_mask;
else
data[i] = 0;
}
}
int main (void)
{
uint32_t program_size = NOR_SECTOR_SIZE;
uint8_t tx_protect_data[program_size] = {0};
uint8_t tx_test_data[program_size] = {0};
uint32_t start_addr = 0x00;
HAL_Init();
/* Configure the system clock */
SystemClock_Config();
AFIO_Init();
if (sn34f2_NORFlash_Init() != HAL_OK)
goto error;
// Before programming, flash need be erased
if(HAL_NORFlash_EraseSector(&NORFlash_Handle, start_addr, HAL_MAX_DELAY) != HAL_OK)
goto error;
// prepare data
fill_data(tx_protect_data, program_size, 1, 8);
// data transmit, programming data this time will succeed
if (HAL_NORFlash_Program(&NORFlash_Handle, start_addr, tx_protect_data, program_size, HAL_MAX_DELAY) != HAL_OK)
{
goto error;
}
memset(rx_data, 0, program_size);
if (HAL_NORFlash_Read(&NORFlash_Handle, NOR_TYPE_RX_READ, start_addr, rx_data, program_size, HAL_MAX_DELAY) != HAL_OK)
{
goto error;
}
for (int i=0; i<program_size; i++)
{
if(rx_data[i] != tx_protect_data[i])
{
goto error;
}
}
// protect tx_protect_data (programmed last time)
if(HAL_NORFlash_ProtectBlock(&NORFlash_Handle, NOR_BLOCK_0, HAL_MAX_DELAY) != HAL_OK)
{
HAL_NORFlash_ProtectResumeAll(&NORFlash_Handle, HAL_MAX_DELAY);
goto error;
}
// prepare data
fill_data(tx_test_data, program_size, 0x60, 8);
// data transmit, programming data this time will not succeed
if(HAL_NORFlash_Program(&NORFlash_Handle, start_addr, tx_test_data, program_size, HAL_MAX_DELAY) != HAL_OK)
{
goto error;
}
memset(rx_data, 0, program_size);
if (HAL_NORFlash_Read(&NORFlash_Handle, NOR_TYPE_RX_READ, start_addr, rx_data, program_size, HAL_MAX_DELAY) != HAL_OK)
{
goto error;
}
for (int i=0; i<program_size; i++)
{
/* Because protecting, the tx_test_data will not be programming, data in flash is tx_protect_data*/
if(rx_data[i] != tx_protect_data[i])
{
goto error;
}
}
HAL_NORFlash_ProtectResumeAll(&NORFlash_Handle, HAL_MAX_DELAY);
while (1)
; //pass
error:
HAL_NORFlash_ProtectResumeAll(&NORFlash_Handle, HAL_MAX_DELAY);
HAL_NORFlash_DeInit(&NORFlash_Handle); // fail
}
