#include "sn34f78x_hal.h"
/* USER CODE BEGIN #0 */
#define WAV_MAX_FRAME 1024
// The Sample function is to read a wav file(test_wav_r.wav) and then write the read content to another wav file(test_wav_w.wav)
HAL_Status_T wav_rw_block(void)
{
uint32_t icnt = 0;
// The file being read
M_FILE_HANDLE_T fh_R = {0};
M_FILE_INIT_T initR = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is wav file, it is FILE_TYPE_WAV
.format = FILE_TYPE_WAV,
// Open File Method
.mode = FILE_OPEN_R,
// The file path, excluding drive letters
.full_path = "test_wav_r.wav",
};
// The file being write
M_FILE_HANDLE_T fh_W = {0};
M_FILE_INIT_T initW = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is wav file, it is FILE_TYPE_WAV
.format = FILE_TYPE_WAV,
// Open File Method
.mode = FILE_OPEN_W,
// The file path, excluding drive letters
.full_path = "test_wav_w.wav",
};
M_FILE_SEEK_T f_seek = {0};
M_FILE_STATUS_T wav_status = {0};
// The file content cache area
M_FILE_INFO_T r_data = {0};
uint8_t *buffer = pvPortMalloc(WAV_MAX_FRAME);
if (buffer == NULL)
{
goto __test_err__;
}
r_data.cache = buffer;
r_data.cache_len = WAV_MAX_FRAME;
// Open file
if (HAL_MFile_OpenFile(&fh_R, &initR) != HAL_OK)
goto __test_err__;
// Get wav file status, it contains:
// 1. wav frame count
// 2. wav frame index
// 3. wav format
// 4. wav channels
// 5. wav bits per sample
// 6. wav sample rate
// 7. wav block align, just for ima pcm
// 8. wav file is end or not
if (HAL_MFile_GetFileStatus(fh_R, &wav_status) != HAL_OK)
goto __test_err__;
// Init wav file when write wav file
initW.wav.audio_format = wav_status.wav.audio_format;
initW.wav.bits_per_sample = wav_status.wav.bits_per_sample;
initW.wav.num_of_channels = wav_status.wav.num_of_channels;
initW.wav.sample_rate = wav_status.wav.sample_rate;
initW.wav.block_align = wav_status.wav.block_align;
if (HAL_MFile_OpenFile(&fh_W, &initW) != HAL_OK)
goto __test_err__;
// Read&Write only 1024 records of data each time, loop to retrieve until the data is read&write
while (fh_R.wav_eof == M_FILE_IS_NOT_EOF)
{
// Read data from file and save content to r_data
if (HAL_MFile_ReadFile(&fh_R, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
// Write data to file
if (HAL_MFile_WriteFile(&fh_W, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
}
if (buffer)
{
vPortFree(buffer);
}
// Close file
if (HAL_MFile_CloseFile(fh_W) != HAL_OK)
goto __test_err__;
// Close file
if (HAL_MFile_CloseFile(fh_R) != HAL_OK)
goto __test_err__;
return HAL_OK;
__test_err__:
return HAL_ERROR;
}
