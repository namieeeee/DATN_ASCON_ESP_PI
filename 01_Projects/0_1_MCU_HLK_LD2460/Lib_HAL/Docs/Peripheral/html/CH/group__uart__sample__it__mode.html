Interrupt Mode
UART Interface » Test Sample

该章节介绍了如何使用 UART 接口在 interrupt 模式下传输数据

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 32
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/* tx complete flag */
static __IO bool tx_flag = false;
/* rx complete flag */
static __IO bool rx_flag = false;
/*****************************************************************************/
void HAL_UART_TxCpltCallback(UART_Handle_T *huart)
{
tx_flag = true;
}
UART_Handle_T uart_tx_handle = {
.instance = SN_UART0,
.init = {
.baud_rate = 115200,
.word_length = UART_WORD_LENGTH_8,
.stop_bits = UART_STOP_BITS_1,
.parity = UART_PARITY_NONE,
},
};
static int32_t _uart_interrupt_transmit(void)
{
if (HAL_UART_Init(&uart_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_Transmit_IT(&uart_tx_handle, data_tx, DATA_LEN) != HAL_OK)
{
return HAL_ERROR;
}
while (tx_flag == false)
{
__NOP();
}
if (HAL_UART_DeInit(&uart_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
void HAL_UART_RxCpltCallback(UART_Handle_T *huart)
{
rx_flag = true;
}
UART_Handle_T uart_rx_handle = {
.instance = SN_UART1,
.init = {
.baud_rate = 115200,
.word_length = UART_WORD_LENGTH_8,
.stop_bits = UART_STOP_BITS_1,
.parity = UART_PARITY_NONE,
},
};
static int32_t _uart_interrupt_receive(void)
{
if (HAL_UART_Init(&uart_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_Receive_IT(&uart_rx_handle, data_rx, DATA_LEN) != HAL_OK)
{
return HAL_ERROR;
}
while (rx_flag == false)
{
__NOP();
}
if (HAL_UART_DeInit(&uart_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_uart_interrupt_transmit();
_uart_interrupt_receive();
while (true)
{
__NOP();
}
}
