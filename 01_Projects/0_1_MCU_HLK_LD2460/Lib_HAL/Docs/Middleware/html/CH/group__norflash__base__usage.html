Base Usage
NORFlash » Usage Flow

Description

使用 NORFlash 模块，可对 Flash 进行交互，除了基本的写/读/擦除操作外，还支持写保护机制、DPD模式

  • Program/Read/Erase
    #include "sn34f78x_hal.h"
    NORFlash_Handle_T NORFlash_Handle;
    /**
    * @brief Program an amount of data to NOR FLASH in blocking mode
    * @param NORFlash_Handle NOR FLASH handle
    * @param startAddr start physical NOR FLASH address
    * @param pData pointer to data buffer
    * @param writeSize amount of data to be programmed
    * @param timeout max waiting time
    * @retval HAL_Status_T
    */
    if (HAL_NORFlash_Program(&NORFlash_Handle, startAddr, pData, writeSize, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Read an amount of data from NOR FLASH in blocking mode
    * @param NORFlash_Handle NOR FLASH handle
    * @param Type RX read type, legal parameter is NOR_TYPE_RX_READ or NOR_TYPE_RX_FAST_READ
    * @param startAddr start physical SPI FLASH address
    * @param pData pointer to data buffer
    * @param readSize amount of data to be read
    * @param timeout timeout duration
    * @retval HAL_Status_T
    */
    if (HAL_NORFlash_Read(&NORFlash_Handle, Type, startAddr, pData, readSize, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Program an amount of data to NOR FLASH in DMA mode
    * @param NORFlash_Handle NOR FLASH handle
    * @param startAddr start physical NOR FLASH address
    * @param pData pointer to data buffer
    * @param writeSize amount of data to be programmed
    * @retval HAL_Status_T
    */
    if (HAL_NORFlash_Program_DMA(&NORFlash_Handle, startAddr, pData, writeSize) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Read an amount of data from NOR FLASH in DMA mode
    * @param NORFlash_Handle NOR FLASH handle
    * @param Type RX read type, legal parameter is NOR_TYPE_RX_READ or NOR_TYPE_RX_FAST_READ
    * @param startAddr start physical NOR FLASH address
    * @param pData pointer to data buffer
    * @param readSize amount of data to be read
    * @retval HAL_Status_T
    */
    if (HAL_NORFlash_Read_DMA(&NORFlash_Handle, Type, startAddr, pData, readSize) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Erase a NOR FLASH sector, one sector size is 4 kilobyte(KB)
    * @param NORFlash_Handle NOR FLASH handle
    * @param startAddr start physical NOR FLASH address
    * @param timeout max waiting time
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_EraseSector(&NORFlash_Handle, start_addr, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
  • Protect
    #include "sn34f78x_hal.h"
    NORFlash_Handle_T NORFlash_Handle;
    /**
    * @brief Protect Block from writing and erasing
    * @param NORFlash_Handle NOR FLASH handle
    * @param blockArea block count to be protect /ref protect_area
    * @param timeout max waiting time
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_ProtectBlock(&NORFlash_Handle, blockArea, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Protect all block from writing and erasing
    * @param NORFlash_Handle NOR FLASH handle
    * @param timeout max waiting time
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_ProtectBlockAll(&NORFlash_Handle, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Resume all block protecting, so it can be programmed
    * @param NORFlash_Handle NOR FLASH handle
    * @param timeout max waiting time
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_ProtectResumeAll(&NORFlash_Handle, timeout) != HAL_OK)
    {
    return HAL_ERROR;
    }
  • Deep Power Down Mode
    #include "sn34f78x_hal.h"
    NORFlash_Handle_T NORFlash_Handle;
    /**
    * @brief Enter DPD, this mode blocks every instruction except Exit DPD
    * @param NORFlash_Handle NOR FLASH handle
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_EnterDPD(&NORFlash_Handle) != HAL_OK)
    {
    return HAL_ERROR;
    }
    /**
    * @brief Exit DPD, return standby mode
    * @param hnor NOR FLASH handle
    * @retval HAL_Status_T
    */
    if(HAL_NORFlash_ExitDPD(&NORFlash_Handle) != HAL_OK)
    {
    return HAL_ERROR;
    }
