Initialize

CAN_Handle_T can0_handle = {
.instance = SN_CAN0,
.init = {
.mode = CAN_MODE_NORMAL,
.pre_scaler = 0,
.time_seg1 = 44,
.time_seg2 = 1,
.sync_jump_width = 0,
.transmit_fifo_priority = HAL_DISABLE,
.auto_retransmission = HAL_ENABLE,
},
};
/*
* 1 initialize the CAN with CAN0
* 2 call CAN0 NVIC_EnableIRQ in file msp_can.c
* 3 implement CAN_IRQHandler in file it_can.c
*/
if (HAL_CAN_Init(&can0_handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_can.c////////////////////////
/**
* @brief CAN MSP Initialization
* @param hcan: CAN handle pointer
* @retval None
*/
void HAL_CAN_MspInit(CAN_Handle_T *hcan)
{
NVIC_ClearPendingIRQ(CAN0_IRQn);
NVIC_EnableIRQ(CAN0_IRQn);
}
////////////////////////it_can.c////////////////////////
/**
* @brief user should implement the CAN0_TX_IRQHandler for handling CAN0 TX IRQ.
* By invoking HAL_CAN_TX_IRQHandler with CAN_Handle, the CAN HAL driver will dispatch the IRQ result
* to weak functions(public in CAN HAL Driver) or callback functions registered by user.
* @return None
*/
__irq void CAN0_TX_IRQHandler(void)
{
/* Call HAL CAN TX IRQ handle */
HAL_CAN_TX_IRQHandler(&can0_handle);
}

Deinitialize

/*
* 1 deinitialize the CAN_Handle
* 2 call CAN0 NVIC_DisableIRQ in file msp_can.c
*/
if (HAL_CAN_DeInit(&CAN_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_can.c////////////////////////
/**
* @brief CAN MSP De-Initialization
* @param hcan: CAN handle pointer
* @retval None
*/
void HAL_CAN_MspDeInit(CAN_Handle_T *hcan)
{
NVIC_ClearPendingIRQ(CAN0_IRQn);
NVIC_DisableIRQ(CAN0_IRQn);
}
