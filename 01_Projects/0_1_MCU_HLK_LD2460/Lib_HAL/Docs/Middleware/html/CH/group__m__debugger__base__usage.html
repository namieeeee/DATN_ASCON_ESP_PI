Debugger » Usage Flow

Description

Debugger 有四种用法: UART，ATCMD，SWO 或者 USB CDC。

Sample Code

  • Initializes
    /**
    * @brief the debugger middleware support three mode print function
    * 1: UART
    * 2: SWO
    */
    /*under uart mode, all fields below should be config */
    Debugger_Handle_T debugger_handle = {
    .init = {
    /* Set debugger mode */
    .debugger_mode = DEBUGGER_MODE_UART,
    /* Set print buffer size */
    .buff_size = 128,
    /* Under DEBUGGER_MODE_UART, set the uart baudrate */
    .baud_rate = 115200,
    /* under DEBUGGER_MODE_UART, set the uart instance */
    .instance = SN_UART0,
    /* under DEBUGGER_MODE_UART, set the end char in AT+CMD and get_chars */
    .cmd_end = '\n',
    /* under DEBUGGER_MODE_UART, set the CMD task priority */
    .monitor_task_priority = 3
    }
    };
    /*initialize the debugger */
    if (HAL_DBG_Init(&debugger_handle) != HAL_OK)
    return HAL_ERROR;

  • AT+CMD
    /**
    * @brief demo AT+CMD
    *
    */
    typedef enum
    {
    TEST1,
    TEST2,
    TEST3,
    } CMD_ID;
    /**
    * If the user registers a custom command line function and enters the correct command line, Debugger will automatically enter the function, and its parameter "cmd" contains the processed information
    * 1. Receive CMD_INFO_T form debugger_handle.cmd_handle.cmd_queue
    * 2. Enter the corresponding function according to "cmd ->cmd_id"
    * 3. Debugger will put the parameters of the command line in "cmd ->param", and the user needs to call the function after handling it by himself
    *
    */
    void dispatch_cmd(void *parameter)
    {
    CMD_INFO_T cmd;
    for (;;)
    {
    HAL_DBG_ReceiveCmd(&console_handle, &cmd);
    switch (cmd.cmd_id)
    {
    case TEST1:
    printf1(cmd.param);
    break;
    case TEST2:
    printf2(cmd.param);
    break;
    case TEST3:
    printf3(cmd.param);
    break;
    }
    }
    }
    /* Register CMD function for printf1 */
    HAL_DBG_Reg_Cmd("Test1", TEST1);
    /* Register CMD function for printf2 */
    HAL_DBG_Reg_Cmd("Test2", TEST2);
    /* Register CMD function for printf3 */
    HAL_DBG_Reg_Cmd("Test3", TEST3);
  • Uninitialize
    /*de-initialize the debugger */
    if (HAL_DBG_DeInit() != HAL_OK)
    return HAL_ERROR;
