#include "sn34f78x_hal.h"
/* Private variables ---------------------------------------------------------*/

__align(4096U) ETH_DMADescTypeDef DMATxDscrTab[ETH_TXBUFNB]; /* Ethernet Tx DMA Descriptor */
__align(4096U) uint8_t Rx_Buff[ETH_RXBUFNB][ETH_RX_BUF_SIZE]; /* Ethernet Receive Buffer */
__align(4096U) uint8_t Tx_Buff[ETH_TXBUFNB][ETH_TX_BUF_SIZE]; /* Ethernet Transmit Buffer */
static char recv_buf[ETH_RX_BUF_SIZE];
/* Global Ethernet handle */
ETH_HandleTypeDef heth;
uint32_t eth_rx_cmp = 0;
/**
* @brief Ethernet Rx Transfer completed callback
* @param heth: ETH handle
* @retval None
*/
void HAL_ETH_RxCpltCallback(ETH_HandleTypeDef *heth)
{
eth_rx_cmp = 1;
}
HAL_Status_T SNX_ETH_RX(void)
{
HAL_Status_T hal_eth_init_status;
__IO ETH_DMADescTypeDef *DmaRxDesc;
uint32_t pkt_len, pkt_cnt = 0;
uint8_t *buffer;
/* Init ETH */
uint8_t MACAddr[6];
heth.Instance = SN_ETHMAC;
heth.Init.AutoNegotiation = ETH_AUTONEGOTIATION_ENABLE;
heth.Init.Speed = ETH_MACCR_SPEED_100M;
heth.Init.DuplexMode = ETH_MODE_FULLDUPLEX;
heth.Init.PhyAddress = DM9162_PHY_ADDRESS;
MACAddr[0] = 0x08;
MACAddr[1] = 0x08;
MACAddr[2] = 0x08;
MACAddr[3] = 0x08;
MACAddr[4] = 0x08;
MACAddr[5] = 0x08;
memcpy(heth.Init.MACAddr, &MACAddr, sizeof(MACAddr));
heth.Init.ChecksumMode = ETH_CHECKSUM_BY_SOFTWARE;
heth.Init.MediaInterface = ETH_MEDIA_INTERFACE_MII;
hal_eth_init_status = HAL_ETH_Init(&heth);
if (hal_eth_init_status != HAL_OK)
{
return hal_eth_init_status;
}
/* Initialize Tx Descriptors list: Chain Mode */
HAL_ETH_DMATxDescListInit(&heth, DMATxDscrTab, &Tx_Buff[0][0], ETH_TXBUFNB);
/* Initialize Rx Descriptors list: Chain Mode */
HAL_ETH_DMARxDescListInit(&heth, DMARxDscrTab, &Rx_Buff[0][0], ETH_RXBUFNB);
/* Enable MAC and DMA transmission and reception */
HAL_ETH_Start(&heth);
while (1)
{
while (!eth_rx_cmp)
{
};
eth_rx_cmp = 0;
pkt_cnt++;
/* get received frame */
if (HAL_ETH_GetReceivedFrame_IT(&heth) != HAL_OK)
return HAL_ERROR;
/* Obtain the size of the packet and put it into the "len" variable. */
pkt_len = heth.RxFrameInfos.length;
buffer = (uint8_t *)heth.RxFrameInfos.buffer;
memcpy(recv_buf, buffer, pkt_len);
/* ReInit Rx Descriptor */
DmaRxDesc = heth.RxFrameInfos.LSRxDesc;
/* release own of this rxdes */
DmaRxDesc->StatusAndVDBC &= ETH_RXDES0_EDORR;
DmaRxDesc->StatusAndVLAN = 0x0;
};
return HAL_OK;
}
