DMA Interface » Test Sample

该章节介绍了如何使用 DMA 接口在中断模式下传输数据。

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
// define interrupt callback flag
volatile int8_t tx_flag = 0;
volatile int8_t rx_flag = 0;
// define data
uint8_t master_out8[TRANSFER_CNT] = {0};
uint8_t slaver_in8[TRANSFER_CNT] = {0};
/**
* @brief define DMA1 Channel 0 as destination, there send the data to SPI TX DATA
*
*/
DMA_Handle_T dma_ch0_m2p = {
.instance = SN_DMA1,
.init = {
.channel = DMA_CHANNEL_0,
.ch_xfer_mode = DMA_MEMORY_TO_PERIPH, // config DMA_MEMORY_TO_PERIPH mode
.ch_priority = DMA_PRIORITY_LOW,
.ch_fifo_th = DMA_FIFO_TH_1,
.ch_sync = HAL_ENABLE,
.src_burst = DMA_SRC_BURST_1,
.src_width = DMA_XFR_WIDTH_8BITS, // do 8bits data transfer
.src_addr_mode = DMA_ADDR_MODE_INC, // increase the source memory address
.src_ahb_master = DMA_AHB_MASTER_PERIPH,
.dst_rs = DMA_RS1_SSP0_TX,
.dst_width = DMA_XFR_WIDTH_8BITS,
.dst_addr_mode = DMA_ADDR_MODE_FIXED, // use SPI_DATA register to do TX, so the address mode config as DMA_ADDR_MODE_FIXED
.dst_ahb_master = DMA_AHB_MASTER_PERIPH,
},
};
/**
* @brief define DMA1 Channel 0 as destination, there send the data to memory
*
*/
DMA_Handle_T dma_ch3_p2m = {
.instance = SN_DMA1,
.init = {
.channel = DMA_CHANNEL_3,
.ch_xfer_mode = DMA_PERIPH_TO_MEMORY, // move the spi rx data to memory
.ch_priority = DMA_PRIORITY_LOW,
.ch_fifo_th = DMA_FIFO_TH_1,
.ch_sync = HAL_ENABLE,
.src_burst = DMA_SRC_BURST_1,
.src_rs = DMA_RS1_SSP1_RX,
.src_width = DMA_XFR_WIDTH_8BITS, // do 8bits data transfer
.src_addr_mode = DMA_ADDR_MODE_FIXED, // current use SPI_DATA register, so the address mode need set to FIXED
.src_ahb_master = DMA_AHB_MASTER_PERIPH,
.dst_width = DMA_XFR_WIDTH_8BITS,
.dst_addr_mode = DMA_ADDR_MODE_INC, // the destination address should increase after one RX
.dst_ahb_master = DMA_AHB_MASTER_PERIPH,
},
};
/**
* @brief let SPI0 work as master, and do TX
*
*/
SPI_Handle_T spi0_master_handle = {
.instance = SN_SPI0,
.init = {
.frame_format = SPI_FFMT_SPI,
.mode = SPI_MODE_MASTER,
.data_size = 8,
.fs_polarity = SPI_FS_POLARITY_LOW,
.clk_polarity = SPI_CLK_POLARITY_LOW,
.clk_phase = SPI_CLK_PHASE_1EDGE,
.sclk_div = 15,
.first_bit = SPI_FIRST_BIT_LSB,
},
};
/**
* @brief let SPI1 work as master, and do RX
*
*/
SPI_Handle_T spi1_slaver_handle = {
.instance = SN_SPI1,
.init = {
.frame_format = SPI_FFMT_SPI,
.mode = SPI_MODE_SLAVE,
.data_size = 8,
.fs_polarity = SPI_FS_POLARITY_LOW,
.clk_polarity = SPI_CLK_POLARITY_LOW,
.clk_phase = SPI_CLK_PHASE_1EDGE,
.sclk_div = 15,
.first_bit = SPI_FIRST_BIT_LSB,
},
};
static HAL_Status_T _spi_init(void)
{
// use SPI0 to initialize the master handler
if (HAL_SPI_Init(&spi0_master_handle) != HAL_OK)
return HAL_ERROR;
// use SPI1 to initialize the slaver handler
if (HAL_SPI_Init(&spi1_slaver_handle) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
static HAL_Status_T _spi_deinit(void)
{
// deinit master if not use spi master handler
if (HAL_SPI_DeInit(&spi0_master_handle) != HAL_OK)
return HAL_ERROR;
// deinit slaver if not use spi slaver handler
if (HAL_SPI_DeInit(&spi1_slaver_handle) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
/*****************************************************************************/
void HAL_SPI_RxCpltCallback(SPI_Handle_T *hspi)
{
rx_flag++;
}
void HAL_SPI_TxCpltCallback(SPI_Handle_T *hspi)
{
tx_flag++;
}
HAL_Status_T _dma_m2p(void)
{
// data transmit, slaver must be started first
if (HAL_SPI_Receive_DMA(&spi1_slaver_handle, slaver_in8, TRANSFER_CNT) != HAL_OK)
return HAL_ERROR;
if (HAL_SPI_Transmit_DMA(&spi0_master_handle, master_out8, TRANSFER_CNT) != HAL_OK)
return HAL_ERROR;
// wait for transmission to complete
while (tx_flag == 0 || rx_flag == 0)
;
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
HAL_Init();
HAL_Status_T ret = HAL_OK;
// the handler must be initialized by HAL_SPI_Init, before any transferring action begin
ret = _spi_init();
if (ret != HAL_OK)
break;
// master tx and slaver rx
ret = _dma_m2p();
if (ret != HAL_OK)
break;
// if the spi is no longer used, HAL_SPI_DeInit is called to release it
ret = _spi_deinit();
if (ret != HAL_OK)
Error_Handler();
while (1)
;
}
