TPM_Handle_T TPM0_Handle = {
.instance = (void *)SN_CT16B0,
.init = {
// Specifies the pre clock
.pre_scaler = 47,
// Specifies the TPM timer/counter mode
.mode = TPM_CTM_TIMER,
// Specifies the TPM counting mode
.cm = TPM_COUNTING_DOWN,
},
};
/*
* 1 initialize the TPM_Handle with CT16B0
* 2 call CT16B0 NVIC_EnableIRQ in file msp_tpm.c
* 3 implement CT16B0_IRQHandler in file it_tpm.c
*/
if (HAL_TPM_Init(&TPM0_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_tpm.c////////////////////////
/**
* @brief TPM MSP Initialization
* @param htpm: TPM handle pointer
* @retval None
*/
void HAL_TPM_MspInit(TPM_Handle_T *htpm)
{
// 1 by HAL_PinMux_Set API setting current IP AFIO pins if need
if (htpm->instance == SN_CT16B0)
{
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_6, GPIO_P26_TPM0_PWM0);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_7, GPIO_P27_TPM0_PWM0N);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_9, GPIO_P29_TPM0_PWM1);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_8, GPIO_P28_TPM0_PWM1N);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_0, GPIO_P00_TPM0_PWM2);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_1, GPIO_P01_TPM0_PWM2N);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_14, GPIO_P214_TPM0_PWM3);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_15, GPIO_P215_TPM0_PWM3N);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_12, GPIO_P112_TPM0_BRK);
HAL_GPIO_SetAFIO(SN_GPIO3, GPIO_PIN_9, GPIO_P39_TPM0_CAP0);
if (__HAL_RCC_CT16B0_IS_CLK_DISABLE())
{
// 2 by SCU register enable current IP clock
__HAL_RCC_CT16B0_CLK_ENABLE();
// 3 by SCU register resetting current IP register to default value
__HAL_RCC_CT16B0_RESET();
}
// 4 by NVIC API enable current IP IRQ
NVIC_ClearPendingIRQ(CT16B0_IRQn);
NVIC_EnableIRQ(CT16B0_IRQn);
}
}
////////////////////////it_tpm.c////////////////////////
/**
* @brief user should implement the CT16B0_IRQHandler for handling CT16B0 IRQ.
* By invoking HAL_TPM_IRQHandler with TPM_Handle, the TPM HAL driver will dispatch the IRQ result
* to weak functions(public in TPM HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void CT16B0_IRQHandler(void)
{
/* Call TPM IRQ handle */
HAL_TPM_IRQHandler(&TPM0_Handle);
}

////////////////////////msp_uart.c////////////////////////
/**
* @brief TPM MSP De-Initialization
* @param huart: TPM handle pointer
* @retval None
*/
void HAL_TPM_MspDeInit(TPM_Handle_T *htpm)
{
if (htpm->instance == SN_CT16B0)
{
// 1 by SCU register disable current IP clock
__HAL_RCC_CT16B0_CLK_DISABLE();
// 2 by NVIC API disable current IP IRQ
NVIC_ClearPendingIRQ(CT16B0_IRQn);
NVIC_DisableIRQ(CT16B0_IRQn);
// 3 by HAL_PinMux_Set API reset current IP AFIO pins to GPIO if need
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_6, GPIO_P26_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_7, GPIO_P27_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_9, GPIO_P29_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_8, GPIO_P28_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_0, GPIO_P00_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_1, GPIO_P01_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_14, GPIO_P214_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO2, GPIO_PIN_15, GPIO_P215_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_12, GPIO_P112_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO3, GPIO_PIN_9, GPIO_P39_GPIO);
}
}
