#include "sn34f78x_hal.h"
HAL_Status_T disk_part_rate(void)
{
FATFS_LOGIC_DEV_T logic;
FATFS_DISK_STATUS_T disk_status = {};
FATFS_DISK_T disk = {
.dev = FATFS_PHY_DEV_SD1,
// FATFS_PART_METHOD_RATE: Partition by rate, unint:1%
.part[0] = {30, FATFS_PART_METHOD_RATE, "dev1"},
.part[1] = {10, FATFS_PART_METHOD_RATE, "dev2"},
// FATFS_PART_METHOD_SECTOR: Partition by sector, unint:sector(512byte)
.part[2] = {40000, FATFS_PART_METHOD_SECTOR, "dev3"},
// FATFS_PART_METHOD_REST: Partition by rest, use all the rest space
.part[3] = {20, FATFS_PART_METHOD_RATE, "dev4"},
};
// SD card partition
if (HAL_MFile_DiskPart(&disk) != HAL_OK)
return HAL_ERROR;
// Get SD card partition status, it contains:
// 1. Physical drive number
// 2. Physical drive Partition index
// 3. Physical drive Sector size (unint:byte)
// 4. Logical drive number, access drive by this number
// 5. Logical drive size (unint:sector)
// 6. Logical drive free size (unint:sector)
// 7. Logical drive label
if (HAL_MFile_GetDiskStatus(&disk_status) != HAL_OK)
return HAL_ERROR;
// Find disk logical drive
if (HAL_MFile_DiskFind(&logic, FATFS_PHY_DEV_SD1, FATFS_LOGIC_PART1) != HAL_OK)
return HAL_ERROR;
if (HAL_MFile_DiskFind(&logic, FATFS_PHY_DEV_SD1, FATFS_LOGIC_PART2) != HAL_OK)
return HAL_ERROR;
if (HAL_MFile_DiskFind(&logic, FATFS_PHY_DEV_SD1, FATFS_LOGIC_PART3) != HAL_OK)
return HAL_ERROR;
if (HAL_MFile_DiskFind(&logic, FATFS_PHY_DEV_SD1, FATFS_LOGIC_PART4) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
