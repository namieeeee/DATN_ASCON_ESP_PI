RTC Interface » Test Sample

该章节介绍了如何使用 RTC 接口设置时间

范例：

#include "sn34f78x_hal.h"
// start time
RTC_Date_T sdate;
RTC_Time_T stime;
// end time
RTC_Date_T edate;
RTC_Time_T etime;
RTC_Handle_T RTC_Handle =
{
.init = {
.clk_source = RTC_CLK_SOURCE_ILRC,
},
};
/*****************************************************************************/
static HAL_Status_T _rtc_set_time_bcd(void)
{
if (HAL_RTC_Init(&RTC_Handle) != HAL_OK)
{
return HAL_ERROR;
}
/* 13:59:59 */
stime.hours = 0x13;
stime.minutes = 0x59;
stime.seconds = 0x59;
if (HAL_RTC_SetTime(&RTC_Handle, &stime, RTC_FORMAT_BCD) != HAL_OK)
{
return HAL_ERROR;
}
/* 2023-4-19 (Wednesday) */
sdate.century = 0x21;
sdate.year = 0x23;
sdate.month = RTC_MONTH_APRIL;
sdate.date = 0x19;
sdate.week_day = RTC_WEEKDAY_WEDNESDAY;
if (HAL_RTC_SetDate(&RTC_Handle, &sdate, RTC_FORMAT_BCD) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
static HAL_Status_T _rtc_get_time_bcd(void)
{
if (HAL_RTC_GetTime(&RTC_Handle, &etime, RTC_FORMAT_BCD) != HAL_OK)
{
return HAL_ERROR;
}
printf("time: %x:%x:%x \r\n", etime.hours, etime.minutes, etime.seconds);
if (HAL_RTC_GetDate(&RTC_Handle, &edate, RTC_FORMAT_BCD) != HAL_OK)
{
return HAL_ERROR;
}
printf("date: %x%x-%x-%x (week:%x) \r\n", edate.century, edate.year, edate.month, edate.date, edate.week_day);
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_rtc_set_time_bcd();
_rtc_get_time_bcd();
while (true)
{
__NOP();
}
}
