Initialize

I2C_Handle_T I2C_Handle = {
/* link to I2C1 IP */
.instance = SN_I2C1,
.init = {
.addressing_mode = I2C_ADDRESSINGMODE_7BIT,
.clock_speed = I2C_CLOCKSPEED_FAST_PLUS,
.gc_en = HAL_DISABLE,
.own_address = 0x7f,
}};
/*
* 1 initialize the I2C with I2C1
* 2 call I2C1 NVIC_EnableIRQ in file msp_i2c.c
* 3 implement I2C1_IRQHandler in file it_i2c.c
*/
if (HAL_I2C_Init(&I2C_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_i2c.c////////////////////////
/**
* @brief I2C MSP Initialization
* @param hI2C: I2C handle pointer
* @retval None
*/
void HAL_I2C_MspInit(I2C_Handle_T *hi2c)
{
NVIC_ClearPendingIRQ(I2C1_IRQn);
NVIC_EnableIRQ(I2C1_IRQn);
}
////////////////////////it_i2c.c////////////////////////
/**
* @brief user should implement the I2C1_IRQHandler for handling I2C1 IRQ.
* By invoking HAL_I2C_IRQHandler with I2C_Handle, the I2C HAL driver will dispatch the IRQ result
* to weak functions(public in I2C HAL Driver) or callback functions registered by user.
* @return void
*/
__irq void I2C1_IRQHandler(void)
{
/* Call I2C IRQ handle */
HAL_I2C_IRQHandler(&I2C_Handle);
}

Deinitialize

/*
* 1 deinitialize the I2C_Handle
* 2 call I2C1 NVIC_DisableIRQ in file msp_i2c.c
*/
if (HAL_I2C_DeInit(&I2C_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_i2c.c////////////////////////
/**
* @brief I2C MSP De-Initialization
* @param hi2c: I2C handle pointer
* @retval None
*/
void HAL_I2C_MspDeInit(I2C_Handle_T *hi2c)
{
NVIC_ClearPendingIRQ(I2C1_IRQn);
NVIC_DisableIRQ(I2C1_IRQn);
}

}

*/

return HAL_ERROR;
