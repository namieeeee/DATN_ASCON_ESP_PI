#include "sn34f78x_hal.h"
#define BUFFER_SIZE 2048
{
/***********file************ */
// File handle
M_FILE_HANDLE_T fh_R = {0};
// File init
M_FILE_INIT_T initR = {
// SD : FATFS_PHY_DEV_SD1
// Flash : FATFS_PHY_DEV_FLASH
// USBD : FATFS_PHY_DEV_USB
.dev = FATFS_PHY_DEV_SD1,
// The file format, For instance, if the file is bin file, it is FILE_TYPE_BIN
.format = FILE_TYPE_BIN,
// Open File Method, read only
.mode = FILE_OPEN_R,
// The file path, excluding drive letters
.full_path = "test_bin_r.bin",
};
// The file being written
M_FILE_HANDLE_T fh_W = {0};
M_FILE_INIT_T initW = {
.dev = FATFS_PHY_DEV_SD1,
.format = FILE_TYPE_BIN,
// Open File Method, write and overwrite
.mode = FILE_OPEN_W,
// The file path, excluding drive letters
.full_path = "test_bin_w.bin",
};
// The file content cache area
M_FILE_INFO_T r_data = {0};
uint8_t *buffer = pvPortMalloc(BIN_SIZE);
if (buffer == NULL)
{
goto __test_err__;
}
r_data.cache = buffer;
r_data.cache_len = BIN_SIZE;
/*************************** */
/***********Directory************ */
//Directory Handle
M_DIR_HANDLE_T dh = {0};
// Directory Init
M_FILE_PATH_T dir = {0};
dir.dev = FATFS_LOGIC_DEV1; // Partition 1
dir.full_path = "test_bin";
// Directory Item information
M_DIR_ITEM_T di = {0};
/******************************** */
}

partition

#include "sn34f78x_hal.h"
{
FATFS_LOGIC_DEV_T logic;
FATFS_DISK_T disk = {
.dev = FATFS_PHY_DEV_SD1,
// FATFS_PART_METHOD_RATE: Partition by rate, unint:1%
.part[0] = {30, FATFS_PART_METHOD_RATE, "dev1"},
.part[1] = {10, FATFS_PART_METHOD_RATE, "dev2"},
// FATFS_PART_METHOD_SECTOR: Partition by sector, unint:sector(512byte)
.part[2] = {40000, FATFS_PART_METHOD_SECTOR, "dev3"},
// FATFS_PART_METHOD_REST: Partition by rest, use all the rest space
.part[3] = {20, FATFS_PART_METHOD_RATE, "dev4"},
};
// Find disk partition 1
if (HAL_MFile_DiskFind(&logic, FATFS_PHY_DEV_SD1, FATFS_LOGIC_PART1) != HAL_OK)
goto __test_err__;
}

seek/read/write

#include "sn34f78x_hal.h"
#define BUFFER_SIZE 2048
{
/***********file************ */
// Get read file infomation, such as size
M_FILE_STATUS_T bin_status = {0};
if (HAL_MFile_GetFileStatus(fh_R, &bin_status) != HAL_OK)
goto __test_err__;
// determine a position where to start reading, if not assigned, start from the beginning by default
M_FILE_SEEK_T f_seek = {0};
f_seek.seek_bin.pos = 0;
if (HAL_MFile_SeekFile(&fh_R, f_seek) != HAL_OK)
goto __test_err__;
// Read and write file
while (fh_R.bin_eof == M_FILE_IS_NOT_EOF)
{
// Read data from file and save content to r_data
if (HAL_MFile_ReadFile(&fh_R, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
// Write data to file
if (HAL_MFile_WriteFile(&fh_W, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
}
/*************************** */
/***********Directory************ */
// Read directory information
if (HAL_MFile_ReadDir(&dh, &di) != HAL_OK)
goto __test_err__;
/******************************** */
}
