PowerManager » Usage Flow

Description

使用 PowerManager 可进入三种低功耗模式: Sleep，DeepSleep，DeepPowerDown。

Sample Code

  • Initializes
    PowerManager_Handle_T hpowermanager;
    static RTC_Handle_T _rtc_hanlde = {
    .init = RTC_CLK_SOURCE_ILRC,
    };
    static PM_RTC_Config_T _alarm_time = {
    .alarm_relative = 10,
    .alarm_absolute = {
    .hours = 0,
    .minutes = 0,
    .seconds = 0,
    },
    };
    PM_GPIO_Config_T alarm_gpio = {
    .pins = GPIO_PIN_14,
    .irq = PM_GPIO_IRQ_BOTH_EDGE,
    };
    if(HAL_OK != HAL_RTC_Init(&_rtc_hanlde))
    {
    return HAL_ERROR;
    }
    /* The low power mode is configured as SLEEP */
    hpowermanager.init.power_mode = PM_SLEEP_MODE;
    /* The wake-up mode is configured as RTC, and the alarm time required for wake-up is configured */
    hpowermanager.init.rtc = _alarm_time;
    /* The wake-up mode is configured to be gpio wake-up, and the specific gpio that triggers the wake-up is configured */
    hpowermanager.init.gpio[0] = alarm_gpio;
    hpowermanager.init.rtc.rtc_handle = &_rtc_hanlde;
    if(HAL_OK != HAL_PowerManager_Init(&hpowermanager))
    {
    return HAL_ERROR;
    }
  • SLEEP MODE
    #include "sn34f78x_hal.h"

    return HAL_ERROR;

    return HAL_ERROR;
