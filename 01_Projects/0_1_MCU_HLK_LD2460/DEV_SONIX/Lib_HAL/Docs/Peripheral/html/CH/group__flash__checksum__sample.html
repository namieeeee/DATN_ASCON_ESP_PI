CheckSum Calculate
Flash Interface » Test Sample

该章节介绍了如何使用 Flash 接口执行 Checksum 计算操作。

范例：

#include "sn34f78x_hal.h"
#define FLASH_USER_START_ADDR 0
#define FLASH_USER_END_ADDR 0x7FFFF
HAL_Status_T FMC_Checksum_Calculate_Sample(void)
{
uint16_t checksum = 0;
if (HAL_FLASHEx_ChecksumCalculate(FLASH_USER_START_ADDR, FLASH_USER_END_ADDR, &checksum) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
int main(void)
{
uint32_t ret = HAL_OK;
/* HAL Driver Init */
HAL_Init();
/* Configure the system clock */
SystemClock_Config();
/* Flash program sample */
ret = FMC_Checksum_Calculate_Sample();
if (ret != HAL_OK)
Error_Handler();
while (1)
;
}
