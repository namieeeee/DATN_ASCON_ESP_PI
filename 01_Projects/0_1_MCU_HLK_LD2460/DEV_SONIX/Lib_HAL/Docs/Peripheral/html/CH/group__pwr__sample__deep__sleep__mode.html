Deep Sleep Mode
PWR Interface » Test Sample

该章节介绍了如何使用PWR进入 Deep Sleep 模式

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
HAL_Status_T _config_wakeup_pin(void)
{
/* Default Enable clock for GPIOx */
__HAL_RCC_GPIO_P3_CLK_ENABLE();
/* Set GPIO 3.4 as the wake up input pin */
GPIO_Init_T gpio_init = {0};
gpio_init.pin = GPIO_PIN_4;
gpio_init.mode = GPIO_MODE_IT_BOTH;
gpio_init.pull = GPIO_PULL_UP;
if (HAL_GPIO_Init(SN_GPIO3, &gpio_init) != HAL_OK)
{
return HAL_ERROR;
}
/* Enable GPIO pin interrupt */
NVIC_ClearPendingIRQ(P3_IRQn);
NVIC_EnableIRQ(P3_IRQn);
return HAL_OK;
}
/*****************************************************************************/
HAL_Status_T _enter_deep_sleep(void)
{
/*At Deep Sleep Mode, GPIO only Non Clock Mode */
if (HAL_GPIO_ChangeClockMode(SN_GPIO3, GPIO_CLK_NON) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_PWR_SetWakeUpSource(PWR_WKPSRC_P3) != HAL_OK)
{
return HAL_ERROR;
}
/* Enter Deep Sleep Mode */
HAL_PWR_EnterDeepSleepMode();
/* Exit eep Sleep Mode, continue running */
return HAL_OK;
}
/*****************************************************************************/
void Error_Handler(void)
{
/* USER CODE BEGIN Error_Handler_Debug */
/* User can add his own implementation to report the HAL error return state */
__disable_irq();
while (1)
{
__NOP();
}
/* USER CODE END Error_Handler_Debug */
}
/*****************************************************************************/
void SystemClock_Config(void)
{
RCC_OscConfig_t OSCCfg = {0};
RCC_ClkConfig_t ClkCfg = {0};
HAL_RCC_GetClockConfig(&ClkCfg);
if(ClkCfg.SYSCLKSource == RCC_SYSCLK_SRC_PLL) {
/* If source of SYSCLK is configured as PLL and want to change any PLL setting(such as NS, FS...) as this time,
Flash clock frequency should be set large enough in the whole sequence of SYSCLK frequency updating */
FLASH_ConfigProgramInitTypeDef CfgInit = {
.OptionType = OPTIONCFG_CLKFREQ,
.CLKFreq = 192,
};
if( HAL_FLASHEx_ConfigProgram(&CfgInit) != HAL_OK)
Error_Handler();
}
/* Configurates Oscillators */
OSCCfg.OscillatorType = RCC_OSC_TYPE_EHS;
OSCCfg.EHSEn = RCC_OSCCLK_CFG_ON;
OSCCfg.PLL.PLLEn = RCC_OSCCLK_CFG_ON;
OSCCfg.PLL.PLLSource = RCC_PLL_SRC_EHS;
OSCCfg.PLL.NS = 64; // 64: EHS_192Mhz; 40: EHS_120Mhz
OSCCfg.PLL.FS = RCC_PLL_DIV4;
if (HAL_RCC_OscConfig(&OSCCfg) != HAL_OK)
{
Error_Handler();
}
/* Initializes the CPU, AHB and APB buses clocks */
ClkCfg.ClockType = RCC_CLK_TYPE_SYSCLK | RCC_CLK_TYPE_HCLK | RCC_CLK_TYPE_APB0CLK | RCC_CLK_TYPE_APB1CLK;
ClkCfg.SYSCLKSource = RCC_SYSCLK_SRC_PLL;
ClkCfg.AHBCLKDivider = RCC_SYSCLK_DIV1;
ClkCfg.APB0CLKDivider = RCC_HCLK_DIV4;
ClkCfg.APB1CLKDivider = RCC_HCLK_DIV2;
if (HAL_RCC_ClockConfig(&ClkCfg) != HAL_OK)
{
Error_Handler();
}
}
/*****************************************************************************/
int32_t main(void)
{
HAL_Init();
/* configure the system clock = 192M*/
SystemClock_Config();
uint32_t ret = HAL_OK;
do
{
/* config P3.4 as wakeup pin & power mode change pin(optional). */
ret = _config_wakeup_pin();
if (ret != HAL_OK)
{
break;
}
/* Key jitter delay */
HAL_Delay(100);
/* suspend systick */
HAL_SuspendTick();
/* enter SLEEP mode */
_enter_deep_sleep();
/* continue running after wakeup and resume systick */
HAL_ResumeTick();
} while (0);
if (ret != HAL_OK)
{
Error_Handler();
}
while (1)
{
__NOP();
}
}
