Sequence Mode
I2C Interface » Test Sample

该章节介绍了如何在 Sequence 模式下使用使用 I2C 接口传输数据.

范例:

第一轮传输完毕，第二轮切换了方向（master从tx切换到rx, slave从rx切换到tx），master必须发送restart.

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 2
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/* master tx complete flag */
static __IO bool m_tx_flag = false;
/* mastrr rx complete flag */
static __IO bool m_rx_flag = false;
/* slaver tx complete flag */
static __IO bool s_tx_flag = false;
/* slaver rx complete flag */
static __IO bool s_rx_flag = false;
/*****************************************************************************/
void HAL_I2C_MasterTxCpltCallback(I2C_Handle_T *hi2c)
{
m_tx_flag = true;
}
void HAL_I2C_SlaveRxCpltCallback(I2C_Handle_T *hi2c)
{
s_rx_flag = true;
}
void HAL_I2C_MasterRxCpltCallback(I2C_Handle_T *hi2c)
{
m_rx_flag = true;
}
void HAL_I2C_SlaveTxCpltCallback(I2C_Handle_T *hi2c)
{
s_tx_flag = true;
}
I2C_Handle_T i2c_tx_handle = {
/* link to I2C0 IP */
.instance = SN_I2C0,
.init = {
.addressing_mode = I2C_ADDRESSINGMODE_7BIT,
.clock_speed = I2C_CLOCKSPEED_FAST_PLUS,
.gc_en = HAL_DISABLE,
.own_address = 0x7f,
},
};
I2C_Handle_T i2c_rx_handle = {
/* link to I2C2 IP */
.instance = SN_I2C2,
.init = {
.addressing_mode = I2C_ADDRESSINGMODE_7BIT,
.clock_speed = I2C_CLOCKSPEED_FAST_PLUS,
.gc_en = HAL_DISABLE,
.own_address = 0x7f,
}};
static int32_t _i2c_init(void)
{
if (HAL_I2C_Init(&i2c_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_Init(&i2c_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
static int32_t _i2c_deinit(void)
{
if (HAL_I2C_DeInit(&i2c_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_DeInit(&i2c_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/**
* @brief Data is sent twice, when the direction is changed(master changes from tx to rx and the slave changes from rx to tx) ,
* then the start command must be resending
* I2C_FIRST_FRAME:send start command
* I2C_FIRST_AND_LAST_FRAME: send start, nack and stop command
*
* @return int32_t
*/
static int32_t _i2c_seq_changedir_tx_2_rx(void)
{
// Without sending nack and stop, send half of the data
if (HAL_I2C_Slave_Seq_Receive_IT(&i2c_rx_handle, data_rx, DATA_LEN / 2, I2C_FIRST_FRAME) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_Master_Seq_Transmit_IT(&i2c_tx_handle, i2c_rx_handle.init.own_address, data_tx, DATA_LEN / 2, I2C_FIRST_FRAME) != HAL_OK)
{
return HAL_ERROR;
}
while (m_tx_flag == false || s_rx_flag == false)
{
__NOP();
}
m_tx_flag = false;
s_rx_flag = false;
// When the master changes from tx to rx and the slave changes from rx to tx, the master must send the restart command
if (HAL_I2C_Slave_Seq_Transmit_IT(&i2c_rx_handle, data_tx + DATA_LEN / 2, DATA_LEN / 2, I2C_FIRST_AND_LAST_FRAME) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_I2C_Master_Seq_Receive_IT(&i2c_tx_handle, i2c_rx_handle.init.own_address, data_rx + DATA_LEN / 2, DATA_LEN / 2, I2C_FIRST_AND_LAST_FRAME) != HAL_OK)
{
return HAL_ERROR;
}
while (s_tx_flag == false || m_rx_flag == false)
{
__NOP();
}
s_tx_flag = false;
m_rx_flag = false;
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_i2c_init();
_i2c_seq_changedir_tx_2_rx();
_i2c_deinit();
while (true)
{
__NOP();
}
}
