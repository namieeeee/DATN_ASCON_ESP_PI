该章节会介绍如何使用 IWDG 接口产出中断

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
#define REFRESH_CNT 5
#define TIMEOUT_CNT 5
static __IO uint8_t refresh_cnt = 0;
static __IO uint8_t timeout_cnt = 0;
/*****************************************************************************/
void HAL_IWDG_TimeoutCallback(IWDG_Handle_T *hiwdg)
{
timeout_cnt++;
}
IWDG_Handle_T iwdg_handle = {
.instance = SN_WDT,
.init.clk_source = IWDG_CLK_SOURCE_ILRC,
.init.mode = IWDG_MODE_IRQ,
.init.timer_constant = 0x000FFFFF,
};
static int32_t _iwdg_interrupt(void)
{
if (HAL_IWDG_Init(&iwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
while (refresh_cnt < REFRESH_CNT)
{
if (iwdg_handle.instance->COUNTER < iwdg_handle->init.timer_constant)
{
if (HAL_IWDG_Refresh(&iwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
refresh_cnt++;
}
}
while (timeout_cnt < TIMEOUT_CNT)
{
__NOP();
}
if (HAL_IWDG_DeInit(&iwdg_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_iwdg_interrupt();
while (true)
{
__NOP();
}
}
