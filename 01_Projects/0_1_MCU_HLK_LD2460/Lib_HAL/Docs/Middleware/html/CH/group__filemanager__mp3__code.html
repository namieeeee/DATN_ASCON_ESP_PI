#include "sn34f78x_hal.h"
/* USER CODE BEGIN #0 */
#define MP3_BUF_SIZE 10240
// The Sample function is to read a mp3 file(test_mp3_r.mp3) and then write the read content to another mp3 file(test_mp3_w.mp3)
HAL_Status_T mp3_rw_block(void)
{
// The file being read
M_FILE_HANDLE_T fh_R = {0};
M_FILE_INIT_T initR = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is mp3 file, it is FILE_TYPE_MP3
.format = FILE_TYPE_MP3,
// Open File Method
.mode = FILE_OPEN_R,
// The file path, excluding drive letters
.full_path = "test_mp3_r.mp3",
};
// The file being write
M_FILE_HANDLE_T fh_W = {0};
M_FILE_INIT_T initW = {
// The SD dev, usually it is FATFS_PHY_DEV_SD1
.dev = FATFS_PHY_DEV_SD1,
// The file format, if the file is mp3 file, it is FILE_TYPE_MP3
.format = FILE_TYPE_MP3,
// Open File Method
.mode = FILE_OPEN_WX,
// The file path, excluding drive letters
.full_path = "test_mp3_w.mp3",
};
M_FILE_SEEK_T f_seek = {0};
M_FILE_STATUS_T mp3_status = {0};
// The file content cache area
M_FILE_INFO_T r_data = {0};
uint8_t *buffer = pvPortMalloc(MP3_BUF_SIZE);
if (buffer == NULL)
goto __test_err__;
r_data.cache = buffer;
r_data.cache_len = MP3_BUF_SIZE;
// Open file
if (HAL_MFile_OpenFile(&fh_R, &initR) != HAL_OK)
goto __test_err__;
// Get mp3 file status, it contains:
// 1. mp3 layer
// 2. mp3 audio version ID
// 3. mp3 bitrate unint: kb/s=1000bit/s
// 4. mp3 sampling rate frequency, unint:hz/s
// 5. mp3 sampling point number
// 6. mp3 channel mode
// 7. mp3 current pos by percent (unint:1%)
// 8. mp3 file is eof or not
if (HAL_MFile_GetFileStatus(fh_R, &mp3_status) != HAL_OK)
goto __test_err__;
// Init mp3 file when write mp3 file
initW.mp3.bit_rate = mp3_status.mp3.bit_rate;
initW.mp3.channel = mp3_status.mp3.channel;
initW.mp3.layer = mp3_status.mp3.layer;
initW.mp3.sample_rate = MP3_SAMPLE_RATE_12000;
initW.mp3.ver = mp3_status.mp3.ver;
if (HAL_MFile_OpenFile(&fh_W, &initW) != HAL_OK)
goto __test_err__;
// Read&Write only one percent of data each time, loop to retrieve until the data is read&write
for(uint16_t i = 0 ; i < 100; i++)
{
f_seek.seek_mp3.seek_ratio = i;
// Seek file
if (HAL_MFile_SeekFile(&fh_R, f_seek) != HAL_OK)
goto __test_err__;
if (HAL_MFile_SeekFile(&fh_W, f_seek) != HAL_OK)
goto __test_err__;
// Read data from file and save content to r_data
if (HAL_MFile_ReadFile(&fh_R, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
// Write data to file
if (HAL_MFile_WriteFile(&fh_W, &r_data, M_FILE_BLOCKING) != HAL_OK)
goto __test_err__;
}
if (buffer)
{
vPortFree(buffer);
}
// Close file
if (HAL_MFile_CloseFile(fh_W) != HAL_OK)
goto __test_err__;
// Close file
if (HAL_MFile_CloseFile(fh_R) != HAL_OK)
goto __test_err__;
return HAL_OK;
__test_err__:
return HAL_ERROR;
}
