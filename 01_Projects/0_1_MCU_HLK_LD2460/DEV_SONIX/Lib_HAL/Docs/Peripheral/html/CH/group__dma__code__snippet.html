/**

*/
DMA_Handle_T dma_ch3_p2m = {
/* link to DMA1 IP */
.instance = SN_DMA1,
.init = {
.channel = DMA_CHANNEL_3,
.ch_xfer_mode = DMA_PERIPH_TO_MEMORY, // move the spi rx data to memory
.ch_priority = DMA_PRIORITY_LOW,
.ch_fifo_th = DMA_FIFO_TH_1,
.ch_sync = HAL_ENABLE,
.src_burst = DMA_SRC_BURST_1,
.src_rs = DMA_RS1_SSP1_RX,
.src_width = DMA_XFR_WIDTH_8BITS, // do 8bits data transfer
.src_addr_mode = DMA_ADDR_MODE_FIXED, // current use SPI_DATA register, so the address mode need set to FIXED
.src_ahb_master = DMA_AHB_MASTER_PERIPH,
.dst_width = DMA_XFR_WIDTH_8BITS,
.dst_addr_mode = DMA_ADDR_MODE_INC, // the destination address should increase after one RX
.dst_ahb_master = DMA_AHB_MASTER_PERIPH,
},
};

////////////////////////hal_msp.c////////////////////////
void HAL_DMA_MspInit(DMA_Handle_T *hdma)
{
if (hdma->instance == SN_DMA1)
{
if (__HAL_RCC_DMA1_IS_CLK_DISABLE())
{
// 1 by SCU register enable current IP clock
__HAL_RCC_DMA1_CLK_ENABLE();
// 2 by SCU register enable current IP clock
__HAL_RCC_DMA1_RESET();
}
// 3 nvic enable irq
if (hdma->init.channel == DMA_CHANNEL_0)
{
NVIC_ClearPendingIRQ(DMA1_CH0_IRQn);
NVIC_EnableIRQ(DMA1_CH0_IRQn);
}
else if (hdma->init.channel == DMA_CHANNEL_3)
{
NVIC_ClearPendingIRQ(DMA1_CH3_IRQn);
NVIC_EnableIRQ(DMA1_CH3_IRQn);
}
}
}
void HAL_SPI_MspInit(SPI_Handle_T *hspi)
{
if (hspi->instance == SN_SPI0)
{
// 1 by HAL_GPIO_SetAFIO API setting current IP AFIO pins if need
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_0, GPIO_P00_SPI_SCK0);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_1, GPIO_P01_SPI_SEL0);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_2, GPIO_P12_SPI_MO0);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_1, GPIO_P11_SPI_MI0);
if (__HAL_RCC_SSP0_IS_CLK_DISABLE())
{
// 2 by SCU register enable current IP clock
__HAL_RCC_SSP0_CLK_ENABLE();
// 3 by SCU register resetting current IP register to default value
__HAL_RCC_SSP0_RESET();
}
// 4 init DMA
if (HAL_DMA_Init(&dma_ch0_m2p) != HAL_OK)
return;
// points the DMA handle to master hdmatx before SPI initialize
HAL_LINKDMA(&spi0_master_handle, hdmatx, dma_ch0_m2p);
}
}

////////////////////////hal_msp.c////////////////////////
void HAL_SPI_MspDeInit(SPI_Handle_T *hspi)
{
if (hspi->instance == SN_SPI0)
{
// 1 by SCU register disable current IP clock
if (__HAL_RCC_SSP0_IS_CLK_ENABLE())
{
__HAL_RCC_SSP0_CLK_DISABLE();
}
// 2 deinit DMA
if (HAL_DMA_DeInit(&dma_ch0_m2p) != HAL_OK)
return;
// 3 by HAL_GPIO_SetAFIO API setting current IP AFIO pins to GPIO
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_0, GPIO_P00_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO0, GPIO_PIN_1, GPIO_P01_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_2, GPIO_P12_GPIO);
HAL_GPIO_SetAFIO(SN_GPIO1, GPIO_PIN_1, GPIO_P11_GPIO);
}
}
void HAL_DMA_MspDeInit(DMA_Handle_T *hdma)
{
if (hdma->instance == SN_DMA1)
{
// 1 get all enable status of current dma
uint32_t all_ch_en = 0;
HAL_DMA_GetChannelStatus(hdma, &all_ch_en);
// 2 by SCU register disable current IP clock while no channel is used
if (__HAL_RCC_DMA1_IS_CLK_ENABLE() && all_ch_en == 0)
{
__HAL_RCC_DMA1_CLK_DISABLE();
}
// 3 nvic disable irq
if (hdma->init.channel == DMA_CHANNEL_3)
{
NVIC_ClearPendingIRQ(DMA1_CH3_IRQn);
NVIC_DisableIRQ(DMA1_CH3_IRQn);
}
}
}
