Initialize

WWDG_Handle_T WWDG_handle = {
.instance = SN_WWDT,
.init.prescaler = WWDG_PRESCALER_4,
.init.counter = 0x10000,
.init.window = 0x5000,
.init.clk_source = WWDG_CLK_SOURCE_ILRC,
.init.mode = WWDG_MODE_OVERFLOW,
};
/*
* 1 initialize the WWDG_handle
* 2 call WWDT_IRQn NVIC_EnableIRQ in file msp_wwdg.c
* 3 implement WWDT_IRQn_IRQHandler in file it_wwdg.c
*/
if (HAL_WWDG_Init(&WWDG_handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_wwdg.c////////////////////////
/**
* @brief WWDG MSP Initialization
* @param hwwdg: WWDG handle pointer
* @retval None
*/
void HAL_WWDG_MspInit(WWDG_Handle_T *hwwdg)
{
NVIC_ClearPendingIRQ(WWDT_IRQn);
NVIC_EnableIRQ(WWDT_IRQn);
}
////////////////////////it_wwdg.c////////////////////////
/**
* @brief user should implement the WWDT_IRQHandler for handling WWDG IRQ.
* By invoking HAL_WWDG_IRQHandle with WWDG_Handle, the WWDG HAL driver will dispatch the IRQ result
* to weak functions(public in WWDG HAL Driver) or callback functions registered by user.
* @retval None
*/
__irq void WWDT__IRQHandler(void)
{
/* Call WWDG IRQ handle */
HAL_WWDG_IRQHandle(&WWDG_handle);
}

Deinitialize

/*
* 1 deinitialize the WWDG_Handle
* 2 call WWDT NVIC_DisableIRQ in file msp_wwdg.c
*/
if (HAL_WWDG_DeInit(&WWDG_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_wwdg.c////////////////////////
/**
* @brief WWDG MSP De-Initialization
* @param hwwdg: WWDG handle pointer
* @retval None
*/
void HAL_WWDG_MspDeInit(WWDG_Handle_T *hwwdg)
{
NVIC_ClearPendingIRQ(WWDT_IRQn);
NVIC_DisableIRQ(WWDT_IRQns);
}
