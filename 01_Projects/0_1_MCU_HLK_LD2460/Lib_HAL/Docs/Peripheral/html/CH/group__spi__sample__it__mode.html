Interrupt Mode
SPI Interface » Test Sample

该章节介绍了如何在中断模式下使用SPI接口传输数据。

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 32
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/* tx complete flag */
static __IO bool tx_flag = false;
/* rx complete flag */
static __IO bool rx_flag = false;
/*****************************************************************************/
void HAL_SPI_TxCpltCallback(SPI_Handle_T *hspi)
{
tx_flag = true;
}
SPI_Handle_T spi_tx_handle = {
.instance = SN_SPI0,
.init = {
.frame_format = SPI_FFMT_SPI,
.mode = SPI_MODE_MASTER,
.data_size = 8,
.fs_polarity = SPI_FS_POLARITY_LOW,
.clk_polarity = SPI_CLK_POLARITY_LOW,
.clk_phase = SPI_CLK_PHASE_1EDGE,
.sclk_div = 10,
.first_bit = SPI_FIRST_BIT_LSB,
},
};
static int32_t _spi_interrupt_transmit(void)
{
if (HAL_SPI_Init(&spi_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_SPI_Transmit_IT(&spi_tx_handle, data_tx, DATA_LEN) != HAL_OK)
{
return HAL_ERROR;
}
while (tx_flag == false)
{
__NOP();
}
if (HAL_SPI_DeInit(&spi_tx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
void HAL_SPI_RxCpltCallback(SPI_Handle_T *hspi)
{
rx_flag = true;
}
SPI_Handle_T spi_rx_handle = {
.instance = SN_SPI1,
.init = {
.frame_format = SPI_FFMT_SPI,
.mode = SPI_MODE_SLAVE,
.data_size = 8,
.fs_polarity = SPI_FS_POLARITY_LOW,
.clk_polarity = SPI_CLK_POLARITY_LOW,
.clk_phase = SPI_CLK_PHASE_1EDGE,
.sclk_div = 10,
.first_bit = SPI_FIRST_BIT_LSB,
},
};
static int32_t _spi_interrupt_receive(void)
{
if (HAL_SPI_Init(&spi_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_SPI_Receive_IT(&spi_rx_handle, data_rx, DATA_LEN) != HAL_OK)
{
return HAL_ERROR;
}
while (rx_flag == false)
{
__NOP();
}
if (HAL_SPI_DeInit(&spi_rx_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_spi_interrupt_transmit();
_spi_interrupt_receive();
while (true)
{
__NOP();
}
}
