Blocking Mode
DMA Interface » Test Sample

该章节介绍了如何使用 DMA 接口在 block 模式下传输数据。

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
// define data
uint8_t memory_out8[TRANSFER_CNT] = {0};
uint8_t memory_in8[TRANSFER_CNT] = {0};
/**
* @brief use DMA1 Channel 0 to move data under DMA_MEMORY_TO_MEMORY
*
*/
DMA_Handle_T hdma_ch0_m2m = {
.instance = SN_DMA1,
.init = {
.channel = DMA_CHANNEL_0,
.ch_xfer_mode = DMA_MEMORY_TO_MEMORY, // config DMA_MEMORY_TO_MEMORY mode
.ch_priority = DMA_PRIORITY_LOW,
.ch_fifo_th = DMA_FIFO_TH_1,
.ch_sync = HAL_ENABLE,
.src_burst = DMA_SRC_BURST_1,
.src_width = DMA_XFR_WIDTH_8BITS, // do 8bits data transfer
.src_addr_mode = DMA_ADDR_MODE_INC, // increase the source memory address
.src_ahb_master = DMA_AHB_MASTER_PERIPH,
.dst_width = DMA_XFR_WIDTH_8BITS,
.dst_addr_mode = DMA_ADDR_MODE_INC, // increase the destination memory address
.dst_ahb_master = DMA_AHB_MASTER_PERIPH,
},
};
/*****************************************************************************/
/**
* @brief master tx and slaver rx.
* @retval HAL status
*/
HAL_Status_T _dma_m2m(void)
{
if (HAL_DMA_Start(&hdma_ch0_m2m, (uint32_t)memory_out8, (uint32_t)memory_in8, TRANSFER_CNT) != HAL_OK)
return HAL_ERROR;
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
HAL_Init();
// do dma handler initialization
ret = HAL_DMA_Init(&hdma_ch0_m2m);
if (ret != HAL_OK)
break;
// do memory to memory transaction by DMA1 channel 0
ret = _dma_m2m();
if (ret != HAL_OK)
break;
// if not use channel 0 of DMA1, by HAL_DMA_DeInit doing deinit
ret = HAL_DMA_DeInit(&hdma_ch0_m2m);
if (ret != HAL_OK)
Error_Handler();
while (1)
;
}
