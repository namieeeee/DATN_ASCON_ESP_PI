SN_GPIO_Type GPIO_Port = SN_GPIO2;
IRQn_Type GPIO_Irq = P2_IRQn;
uint32_t GPIO_Pin = GPIO_PIN_1;
GPIO_Init_T GPIO_Init = {
.pin = GPIO_PIN_1,
.mode = GPIO_MODE_OUTPUT,
};
/*
* 1 initialize the GPIO with GPIO2
* 2 call GPIO2 NVIC_EnableIRQ
* 3 implement GPIO2_IRQHandler in file it_gpio.c
*/
if (HAL_GPIO_Init(GPIO_Port, &GPIO_Init) != HAL_OK)
{
return HAL_ERROR;
}
NVIC_EnableIRQ(GPIO_Irq);
////////////////////////it_gpio.c////////////////////////
/**
* @brief user should implement the GPIO2_IRQHandler for handling GPIO2 IRQ.
* By invoking HAL_GPIO_IRQHandler with GPIO_Handle, the GPIO HAL driver will dispatch the IRQ result
* to weak functions(public in GPIO HAL Driver) or callback functions registered by user.
* @return None
*/
__irq void GPIO2_IRQHandler(void)
{
/* Call GPIO IRQ handle */
HAL_GPIO_IRQHandler(GPIO_Port);
}

* 1 deinitialize the GPIO pin

* 2 call GPIO2 NVIC_DisableIRQ
*/
