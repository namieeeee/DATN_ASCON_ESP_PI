Initialize

SPI_Handle_T SPI_Handle = {
.instance = SN_SPI0,
.init = {
.frame_format = SPI_FFMT_SPI,
.mode = SPI_MODE_MASTER,
.data_size = 8,
.fs_polarity = SPI_FS_POLARITY_LOW,
.clk_polarity = SPI_CLK_POLARITY_LOW,
.clk_phase = SPI_CLK_PHASE_1EDGE,
.sclk_div = 10,
.first_bit = SPI_FIRST_BIT_LSB,
},
};
/*
* 1 initialize the SPI with SPI0
* 2 call SPI0 NVIC_EnableIRQ in file msp_spi.c
* 3 implement SPI_IRQHandler in file it_spi.c
*/
if (HAL_SPI_Init(&SPI_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_spi.c////////////////////////
/**
* @brief SPI MSP Initialization
* @param hspi: SPI handle pointer
* @retval None
*/
void HAL_SPI_MspInit(SPI_Handle_T *hspi)
{
NVIC_ClearPendingIRQ(SPI0_IRQn);
NVIC_EnableIRQ(SPI0_IRQn);
}
////////////////////////it_spi.c////////////////////////
/**
* @brief user should implement the SPI0_IRQHandler for handling SPI0 IRQ.
* By invoking HAL_SPI_IRQHandler with SPI_Handle, the SPI HAL driver will dispatch the IRQ result
* to weak functions(public in SPI HAL Driver) or callback functions registered by user.
* @return None
*/
__irq void SPI0_IRQHandler(void)
{
/* Call HAL SPI IRQ handle */
HAL_SPI_IRQHandler(&SPI_Handle);
}

Deinitialize

/*
* 1 deinitialize the SPI_Handle
* 2 call SPI0 NVIC_DisableIRQ in file msp_spi.c
*/
if (HAL_SPI_Init(&SPI_Handle) != HAL_OK)
{
return HAL_ERROR;
}
////////////////////////msp_spi.c////////////////////////
/**
* @brief SPI MSP De-Initialization
* @param hspi: SPI handle pointer
* @retval None
*/
void HAL_SPI_MspDeInit(SPI_Handle_T *hspi)
{
NVIC_ClearPendingIRQ(SPI0_IRQn);
NVIC_DisableIRQ(SPI0_IRQn);
}
