Blocking Mode
UART Interface » Test Sample

该章节介绍了如何使用 UART 接口在 Blocking 模式下传输数据

范例：

#include "sn34f78x_hal.h"
/*****************************************************************************/
/* data buffer size */
#define DATA_LEN 32
/* data buffer for transmitting data */
static uint8_t data_tx[DATA_LEN];
/* data buffer for receiving data */
static uint8_t data_rx[DATA_LEN];
/* timeout ticks */
static uint32_t timeout = HAL_MAX_DELAY;
/*****************************************************************************/
static int32_t _uart_blocking_transmit(void)
{
UART_Handle_T uart_handle = {
.instance = SN_UART0,
.init = {
.baud_rate = 115200,
.word_length = UART_WORD_LENGTH_8,
.stop_bits = UART_STOP_BITS_1,
.parity = UART_PARITY_NONE,
},
};
if (HAL_UART_Init(&uart_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_Transmit(&uart_handle, data_tx, DATA_LEN, timeout) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_DeInit(&uart_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
static int32_t _uart_blocking_receive(void)
{
UART_Handle_T uart_handle = {
.instance = SN_UART1,
.init = {
.baud_rate = 115200,
.word_length = UART_WORD_LENGTH_8,
.stop_bits = UART_STOP_BITS_1,
.parity = UART_PARITY_NONE,
},
};
if (HAL_UART_Init(&uart_handle) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_Receive(&uart_handle, data_rx, DATA_LEN, timeout) != HAL_OK)
{
return HAL_ERROR;
}
if (HAL_UART_DeInit(&uart_handle) != HAL_OK)
{
return HAL_ERROR;
}
return HAL_OK;
}
/*****************************************************************************/
int32_t main(void)
{
_uart_blocking_transmit();
_uart_blocking_receive();
while (true)
{
__NOP();
}
}
